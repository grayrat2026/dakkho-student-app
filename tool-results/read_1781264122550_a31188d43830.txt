    29тЖТ    29тЖТfunction getVideoDuration(file: File): Promise<number> {
    30тЖТ    30тЖТ  return new Promise((resolve) => {
    31тЖТ    31тЖТ    const video = document.createElement('video');
    32тЖТ    32тЖТ    video.preload = 'metadata';
    33тЖТ    33тЖТ    video.onloadedmetadata = () => {
    34тЖТ    34тЖТ      resolve(Math.round(video.duration));
    35тЖТ    35тЖТ      URL.revokeObjectURL(video.src);
    36тЖТ    36тЖТ    };
    37тЖТ    37тЖТ    video.onerror = () => resolve(0);
    38тЖТ    38тЖТ    video.src = URL.createObjectURL(file);
    39тЖТ    39тЖТ  });
    40тЖТ    40тЖТ}
