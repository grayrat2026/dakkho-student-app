     1→'use client';
     2→
     3→import { useState, useMemo, useRef, useCallback } from 'react';
     4→import { motion, AnimatePresence } from 'framer-motion';
     5→import {
     6→  ArrowLeft, Plus, ChevronDown, ChevronRight, Trash2, Edit3,
     7→  FolderOpen, Video, FileText, Link as LinkIcon, Radio,
     8→  BookOpen, Play, Eye, X, Save, MessageSquare, Send,
     9→  Upload, Search, Paperclip, File, Check,
    10→} from 'lucide-react';
    11→import { toast } from 'sonner';
    12→import { GlassCard } from '@/components/shared/GlassCard';
    13→import { GradientButton } from '@/components/shared/GradientButton';
    14→import { LoadingSkeleton } from '@/components/shared/LoadingSkeleton';
    15→import { EmptyState } from '@/components/shared/EmptyState';
    16→import {
    17→  useCurriculum, useCourseSubjects, useCourse,
    18→  useCreateChapter, useUpdateChapter, useDeleteChapter,
    19→  useCreateLesson, useUpdateLesson, useDeleteLesson,
    20→  useReviews, useReplyReview,
    21→} from '@/lib/api-hooks';
    22→import { useNavigationStore } from '@/lib/store';
    23→import { cn, formatDurationShort, relativeTime } from '@/lib/utils';
    24→import { staggerChildren, collapseExpand } from '@/lib/animations';
    25→import { apiGet, apiUpload } from '@/lib/api-client';
    26→
    27→type AddMode = 'chapter' | 'lesson' | null;
    28→type EditMode = 'chapter' | 'lesson' | null;
    29→
    30→export function CourseSubject() {
    31→  const params = useNavigationStore((s) => s.params);
    32→  const navigate = useNavigationStore((s) => s.navigate);
    33→  const goBack = useNavigationStore((s) => s.goBack);
    34→  const courseId = params.courseId || '';
    35→  const subjectId = params.subjectId || '';  // Can be UUID string or number
    36→
    37→  const { data: courseData, loading: courseLoading, error: courseError } = useCourse(courseId);
    38→  const { data: subjectsData, loading: subjectsLoading } = useCourseSubjects(courseId);
    39→  const { data: curriculumData, loading: curriculumLoading, refetch: refetchCurriculum } = useCurriculum(courseId);
    40→  const { mutate: createChapter, loading: creatingChapter } = useCreateChapter(courseId);
    41→  const { mutate: updateChapter, loading: updatingChapter } = useUpdateChapter(courseId);
    42→  const { mutate: deleteChapter, loading: deletingChapter } = useDeleteChapter(courseId);
    43→  const { mutate: createLesson, loading: creatingLesson } = useCreateLesson(courseId);
    44→  const { mutate: updateLesson, loading: updatingLesson } = useUpdateLesson(courseId);
    45→  const { mutate: deleteLesson, loading: deletingLesson } = useDeleteLesson(courseId);
    46→
    47→  // Reviews for this course
    48→  const { data: reviewsData, refetch: refetchReviews } = useReviews(1);
    49→  const { mutate: replyReview, loading: replyingReview } = useReplyReview();
    50→
    51→  const course = courseData?.course || courseData;
    52→  const allSubjects = subjectsData?.subjects || [];
    53→  const currentSubject = allSubjects.find((s: any) => String(s.subjectId || s.id) === String(subjectId));
    54→  const curriculum = curriculumData?.curriculum || { subjects: [], chapters: [], lessons: [], resources: [] };
    55→
    56→  // Filter data for this subject
    57→  const subjectChapters = useMemo(() => {
    58→    return (curriculum.chapters || [])
    59→      .filter((ch: any) => String(ch.subjectId || ch.subject_id) === String(subjectId))
    60→      .sort((a: any, b: any) => (a.sortOrder || a.sort_order || 0) - (b.sortOrder || b.sort_order || 0));
    61→  }, [curriculum, subjectId]);
    62→
    63→  const allLessons = useMemo(() => {
    64→    return (curriculum.lessons || [])
    65→      .filter((l: any) => String(l.subjectId || l.subject_id) === String(subjectId));
    66→  }, [curriculum, subjectId]);
    67→
    68→  const reviews = useMemo(() => reviewsData?.reviews || [], [reviewsData]);
    69→
    70→  // State
    71→  const [expandedChapters, setExpandedChapters] = useState<Set<string>>(new Set());
    72→  const [addMode, setAddMode] = useState<AddMode>(null);
    73→  const [selectedChapterId, setSelectedChapterId] = useState<string>('');
    74→
    75→  // Chapter add form
    76→  const [chapterTitle, setChapterTitle] = useState('');
    77→  const [chapterDesc, setChapterDesc] = useState('');
    78→
    79→  // Lesson add form
    80→  const [lessonTitle, setLessonTitle] = useState('');
    81→  const [lessonType, setLessonType] = useState<'video' | 'link' | 'youtube' | 'document'>('video');
    82→  const [lessonVideoUrl, setLessonVideoUrl] = useState('');
    83→  const [lessonDesc, setLessonDesc] = useState('');
    84→  const [lessonIsPreview, setLessonIsPreview] = useState(false);
    85→
    86→  // Edit state
    87→  const [editMode, setEditMode] = useState<EditMode>(null);
    88→  const [editId, setEditId] = useState<string>('');
    89→
    90→  // Chapter edit form
    91→  const [editChapterTitle, setEditChapterTitle] = useState('');
    92→  const [editChapterDesc, setEditChapterDesc] = useState('');
    93→
    94→  // Lesson edit form
    95→  const [editLessonTitle, setEditLessonTitle] = useState('');
    96→  const [editLessonType, setEditLessonType] = useState<'video' | 'link' | 'youtube' | 'document'>('video');
    97→  const [editLessonVideoUrl, setEditLessonVideoUrl] = useState('');
    98→  const [editLessonDesc, setEditLessonDesc] = useState('');
    99→  const [editLessonIsPreview, setEditLessonIsPreview] = useState(false);
   100→
   101→  // Attachment / Document upload state
   102→  const [lessonDocument, setLessonDocument] = useState<File | null>(null);
   103→  const [lessonDocumentName, setLessonDocumentName] = useState<string>('');
   104→  const [uploadingDocument, setUploadingDocument] = useState(false);
   105→  const fileInputRef = useRef<HTMLInputElement>(null);
   106→
   107→  // Video search state
   108→  const [videoSearchQuery, setVideoSearchQuery] = useState('');
   109→  const [videoSearchResults, setVideoSearchResults] = useState<any[]>([]);
   110→  const [searchingVideos, setSearchingVideos] = useState(false);
   111→  const [selectedExistingVideo, setSelectedExistingVideo] = useState<any>(null);
   112→  const [showVideoSearch, setShowVideoSearch] = useState(false);
   113→
   114→  // Reply form
   115→  const [replyTexts, setReplyTexts] = useState<Record<number, string>>({});
   116→
   117→  const toggleChapter = (chapterId: string) => {
   118→    setExpandedChapters((prev) => {
   119→      const next = new Set(prev);
   120→      if (next.has(chapterId)) next.delete(chapterId);
   121→      else next.add(chapterId);
   122→      return next;
   123→    });
   124→  };
   125→
   126→  const startEditChapter = (chapter: any) => {
   127→    setEditMode('chapter');
   128→    setEditId(chapter.id);
   129→    setEditChapterTitle(chapter.title || '');
   130→    setEditChapterDesc(chapter.description || '');
   131→    setAddMode(null);
   132→  };
   133→
   134→  const startEditLesson = (lesson: any) => {
   135→    setEditMode('lesson');
   136→    setEditId(lesson.id);
   137→    setEditLessonTitle(lesson.title || '');
   138→    setEditLessonType((lesson.lessonType || lesson.lesson_type || 'video') as any);
   139→    setEditLessonVideoUrl(lesson.videoUrl || lesson.video_url || '');
   140→    setEditLessonDesc(lesson.description || '');
   141→    setEditLessonIsPreview(!!lesson.isPreview);
   142→    setAddMode(null);
   143→  };
   144→
   145→  const cancelEdit = () => {
   146→    setEditMode(null);
   147→    setEditId('');
   148→  };
   149→
   150→  const handleCreateChapter = async () => {
   151→    if (!chapterTitle.trim()) { toast.error('Enter chapter title'); return; }
   152→    try {
   153→      const result = await createChapter({
   154→        title: chapterTitle.trim(),
   155→        subjectId,
   156→        description: chapterDesc.trim() || undefined,
   157→        sortOrder: subjectChapters.length + 1,
   158→      });
   159→      if (result) {
   160→        toast.success('Chapter created');
   161→        setChapterTitle('');
   162→        setChapterDesc('');
   163→        setAddMode(null);
   164→        refetchCurriculum();
   165→      } else {
   166→        toast.error('Failed to create chapter');
   167→      }
   168→    } catch (err: any) {
   169→      toast.error(err?.message || 'Failed to create chapter');
   170→    }
   171→  };
   172→
   173→  const handleUpdateChapter = async () => {
   174→    if (!editChapterTitle.trim()) { toast.error('Enter chapter title'); return; }
   175→    try {
   176→      const result = await updateChapter({
   177→        chapterId: editId,
   178→        title: editChapterTitle.trim(),
   179→        description: editChapterDesc.trim() || undefined,
   180→        subjectId,
   181→      });
   182→      if (result) {
   183→        toast.success('Chapter updated');
   184→        setEditMode(null);
   185→        setEditId('');
   186→        refetchCurriculum();
   187→      } else {
   188→        toast.error('Failed to update chapter');
   189→      }
   190→    } catch (err: any) {
   191→      toast.error(err?.message || 'Failed to update chapter');
   192→    }
   193→  };
   194→
   195→  const handleDeleteChapter = async (chapterId: string) => {
   196→    if (!confirm('Delete this chapter and all its lessons?')) return;
   197→    try {
   198→      const result = await deleteChapter({ chapterId });
   199→      if (result !== null) {
   200→        toast.success('Chapter deleted');
   201→        refetchCurriculum();
   202→      } else {
   203→        toast.error('Failed to delete chapter');
   204→      }
   205→    } catch (err: any) {
   206→      toast.error(err?.message || 'Failed to delete chapter');
   207→    }
   208→  };
   209→
   210→  const handleCreateLesson = async () => {
   211→    if (!lessonTitle.trim()) { toast.error('Enter lesson title'); return; }
   212→    if (!selectedChapterId) { toast.error('Select a chapter'); return; }
   213→    try {
   214→      // If an existing video was selected, use its URL
   215→      let videoUrl = lessonVideoUrl.trim();
   216→      if (selectedExistingVideo) {
   217→        videoUrl = selectedExistingVideo.videoUrl || selectedExistingVideo.video_url || '';
   218→      }
   219→
   220→      // Upload document/PDF if attached
   221→      let documentUrl: string | undefined;
   222→      if (lessonDocument) {
   223→        setUploadingDocument(true);
   224→        try {
   225→          const fd = new FormData();
   226→          fd.append('file', lessonDocument);
   227→          fd.append('bucket', 'resources');
   228→          fd.append('prefix', `courses/${courseId}/documents`);
   229→          const uploadResult = await apiUpload('/upload', fd);
   230→          documentUrl = uploadResult?.url || uploadResult?.key || uploadResult?.publicUrl || '';
   231→          if (!documentUrl && uploadResult) {
   232→            // Try to construct public URL from the response
   233→            const key = uploadResult.key || uploadResult.fileName || '';
   234→            if (key) documentUrl = `https://dakkho-admin-api.dakkho-admin.workers.dev/upload/resources/${key}`;
   235→          }
   236→          toast.success('Document uploaded');
   237→        } catch (docErr: any) {
   238→          console.error('Document upload error:', docErr);
   239→          toast.warning('Document upload failed. Lesson will be created without the document.');
   240→        }
   241→        setUploadingDocument(false);
   242→      }
   243→
   244→      const result = await createLesson({
   245→        title: lessonTitle.trim(),
   246→        chapterId: selectedChapterId,
   247→        subjectId,
   248→        lessonType,
   249→        videoUrl: videoUrl || undefined,
   250→        description: lessonDesc.trim() || undefined,
   251→        isPreview: lessonIsPreview,
   252→        sortOrder: (allLessons.filter((l: any) => (l.chapterId || l.chapter_id) === selectedChapterId).length) + 1,
   253→        documentUrl: documentUrl || undefined,
   254→      });
   255→      if (result) {
   256→        toast.success('Lesson added');
   257→        setLessonTitle('');
   258→        setLessonVideoUrl('');
   259→        setLessonDesc('');
   260→        setLessonIsPreview(false);
   261→        setLessonDocument(null);
   262→        setLessonDocumentName('');
   263→        setSelectedExistingVideo(null);
   264→        setShowVideoSearch(false);
   265→        setAddMode(null);
   266→        refetchCurriculum();
   267→      } else {
   268→        toast.error('Failed to add lesson');
   269→      }
   270→    } catch (err: any) {
   271→      toast.error(err?.message || 'Failed to add lesson');
   272→    }
   273→  };
   274→
   275→  const handleUpdateLesson = async () => {
   276→    if (!editLessonTitle.trim()) { toast.error('Enter lesson title'); return; }
   277→    try {
   278→      const result = await updateLesson({
   279→        lessonId: editId,
   280→        title: editLessonTitle.trim(),
   281→        description: editLessonDesc.trim() || undefined,
   282→        lessonType: editLessonType,
   283→        videoUrl: editLessonVideoUrl.trim() || undefined,
   284→        isPreview: editLessonIsPreview,
   285→      });
   286→      if (result) {
   287→        toast.success('Lesson updated');
   288→        setEditMode(null);
   289→        setEditId('');
   290→        refetchCurriculum();
   291→      } else {
   292→        toast.error('Failed to update lesson');
   293→      }
   294→    } catch (err: any) {
   295→      toast.error(err?.message || 'Failed to update lesson');
   296→    }
   297→  };
   298→
   299→  const handleDeleteLesson = async (lessonId: string) => {
   300→    if (!confirm('Delete this lesson?')) return;
   301→    try {
   302→      const result = await deleteLesson({ lessonId });
   303→      if (result !== null) {
   304→        toast.success('Lesson deleted');
   305→        refetchCurriculum();
   306→      } else {
   307→        toast.error('Failed to delete lesson');
   308→      }
   309→    } catch (err: any) {
   310→      toast.error(err?.message || 'Failed to delete lesson');
   311→    }
   312→  };
   313→
   314→  const handleReply = async (reviewId: number) => {
   315→    const text = replyTexts[reviewId]?.trim();
   316→    if (!text) return;
   317→    try {
   318→      const result = await replyReview({ reviewId, replyText: text });
   319→      if (result) {
   320→        toast.success('Reply posted');
   321→        setReplyTexts((prev) => ({ ...prev, [reviewId]: '' }));
   322→        refetchReviews();
   323→      } else {
   324→        toast.error('Failed to post reply');
   325→      }
   326→    } catch (err: any) {
   327→      toast.error(err?.message || 'Failed to post reply');
   328→    }
   329→  };
   330→
   331→  // Expand all chapters by default
   332→  useMemo(() => {
   333→    if (subjectChapters.length > 0 && expandedChapters.size === 0) {
   334→      setExpandedChapters(new Set(subjectChapters.map((ch: any) => ch.id)));
   335→    }
   336→  }, [subjectChapters.length]); // eslint-disable-line react-hooks/exhaustive-deps
   337→
   338→  if (courseLoading || subjectsLoading || curriculumLoading) {
   339→    return <LoadingSkeleton type="card" count={3} />;
   340→  }
   341→
   342→  if (courseError) {
   343→    return (
   344→      <div className="flex flex-col items-center justify-center py-20 space-y-4">
   345→        <p className="text-sm text-red-500">Failed to load course: {courseError}</p>
   346→        <GradientButton size="sm" onClick={() => refetchCurriculum()}>Retry</GradientButton>
   347→      </div>
   348→    );
   349→  }
   350→
   351→  if (!currentSubject) {
   352→    return (
   353→      <EmptyState
   354→        icon={BookOpen}
   355→        title="Subject not found"
   356→        description="This subject doesn't exist or has been removed."
   357→        actionLabel="Back to Curriculum"
   358→        onAction={() => navigate('course-curriculum', { courseId })}
   359→      />
   360→    );
   361→  }
   362→
   363→  const lessonTypeIcon: Record<string, typeof Video> = {
   364→    video: Video, link: LinkIcon, youtube: Video, document: FileText,
   365→  };
   366→  const lessonTypeLabel: Record<string, string> = {
   367→    video: 'Video', link: 'Link', youtube: 'YouTube', document: 'Document/PDF',
   368→  };
   369→  const lessonTypeColor: Record<string, string> = {
   370→    video: 'bg-neutral-100 text-neutral-700 dark:bg-neutral-800/30 dark:text-neutral-300',
   371→    link: 'bg-blue-100 text-blue-700 dark:bg-blue-900/30 dark:text-blue-400',
   372→    youtube: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   373→    document: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
   374→  };
   375→
   376→  return (
   377→    <div className="space-y-6">
   378→      {/* Header with Back */}
   379→      <motion.div initial={{ opacity: 0, y: 8 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }}>
   380→        <div className="flex items-center gap-3">
   381→          <button
   382→            onClick={() => navigate('course-curriculum', { courseId })}
   383→            className="w-10 h-10 rounded-xl bg-black/[0.03] dark:bg-white/[0.03] flex items-center justify-center text-muted-foreground hover:text-foreground hover:bg-black/[0.06] dark:hover:bg-white/[0.06] transition-colors"
   384→          >
   385→            <ArrowLeft className="w-5 h-5" />
   386→          </button>
   387→          <div className="flex-1 min-w-0">
   388→            <h1 className="text-2xl md:text-3xl font-extrabold text-foreground truncate">
   389→              {currentSubject.subjectName || currentSubject.subject_name || currentSubject.name || 'Subject'}
   390→            </h1>
   391→            <p className="text-muted-foreground mt-0.5 text-sm">{course?.title || ''}</p>
   392→          </div>
   393→          <GradientButton
   394→            size="sm"
   395→            icon={<Plus className="w-4 h-4" />}
   396→            onClick={() => { setAddMode('chapter'); setSelectedChapterId(''); setEditMode(null); }}
   397→          >
   398→            Chapter
   399→          </GradientButton>
   400→          <GradientButton
   401→            size="sm"
   402→            variant="ghost"
   403→            icon={<Plus className="w-4 h-4" />}
   404→            onClick={() => { setAddMode('lesson'); setSelectedChapterId(''); setEditMode(null); }}
   405→          >
   406→            Lesson
   407→          </GradientButton>
   408→        </div>
   409→      </motion.div>
   410→
   411→      {/* Add Chapter Form */}
   412→      <AnimatePresence>
   413→        {addMode === 'chapter' && (
   414→          <motion.div {...collapseExpand}>
   415→            <GlassCard className="p-5" glow>
   416→              <div className="flex items-center justify-between mb-4">
   417→                <h3 className="text-base font-bold gradient-text">New Chapter</h3>
   418→                <button onClick={() => setAddMode(null)} className="w-8 h-8 rounded-lg bg-muted/50 flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors">
   419→                  <X className="w-4 h-4" />
   420→                </button>
   421→              </div>
   422→              <div className="space-y-3">
   423→                <div>
   424→                  <label className="text-sm font-semibold text-foreground mb-1 block">Chapter Title *</label>
   425→                  <input
   426→                    type="text"
   427→                    value={chapterTitle}
   428→                    onChange={(e) => setChapterTitle(e.target.value)}
   429→                    placeholder="Enter chapter title"
   430→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60"
   431→                    autoFocus
   432→                  />
   433→                </div>
   434→                <div>
   435→                  <label className="text-sm font-semibold text-foreground mb-1 block">Description</label>
   436→                  <textarea
   437→                    value={chapterDesc}
   438→                    onChange={(e) => setChapterDesc(e.target.value)}
   439→                    placeholder="Optional chapter description"
   440→                    rows={2}
   441→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60 resize-none"
   442→                  />
   443→                </div>
   444→                <div className="flex justify-end gap-2">
   445→                  <GradientButton variant="ghost" size="sm" onClick={() => setAddMode(null)}>Cancel</GradientButton>
   446→                  <GradientButton size="sm" onClick={handleCreateChapter} loading={creatingChapter} disabled={!chapterTitle.trim()} icon={<Save className="w-3.5 h-3.5" />}>
   447→                    Create Chapter
   448→                  </GradientButton>
   449→                </div>
   450→              </div>
   451→            </GlassCard>
   452→          </motion.div>
   453→        )}
   454→      </AnimatePresence>
   455→
   456→      {/* Edit Chapter Form */}
   457→      <AnimatePresence>
   458→        {editMode === 'chapter' && (
   459→          <motion.div {...collapseExpand}>
   460→            <GlassCard className="p-5" glow>
   461→              <div className="flex items-center justify-between mb-4">
   462→                <h3 className="text-base font-bold gradient-text">Edit Chapter</h3>
   463→                <button onClick={cancelEdit} className="w-8 h-8 rounded-lg bg-muted/50 flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors">
   464→                  <X className="w-4 h-4" />
   465→                </button>
   466→              </div>
   467→              <div className="space-y-3">
   468→                <div>
   469→                  <label className="text-sm font-semibold text-foreground mb-1 block">Chapter Title *</label>
   470→                  <input
   471→                    type="text"
   472→                    value={editChapterTitle}
   473→                    onChange={(e) => setEditChapterTitle(e.target.value)}
   474→                    placeholder="Enter chapter title"
   475→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60"
   476→                    autoFocus
   477→                  />
   478→                </div>
   479→                <div>
   480→                  <label className="text-sm font-semibold text-foreground mb-1 block">Description</label>
   481→                  <textarea
   482→                    value={editChapterDesc}
   483→                    onChange={(e) => setEditChapterDesc(e.target.value)}
   484→                    placeholder="Optional chapter description"
   485→                    rows={2}
   486→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60 resize-none"
   487→                  />
   488→                </div>
   489→                <div className="flex justify-end gap-2">
   490→                  <GradientButton variant="ghost" size="sm" onClick={cancelEdit}>Cancel</GradientButton>
   491→                  <GradientButton size="sm" onClick={handleUpdateChapter} loading={updatingChapter} disabled={!editChapterTitle.trim()} icon={<Save className="w-3.5 h-3.5" />}>
   492→                    Save Changes
   493→                  </GradientButton>
   494→                </div>
   495→              </div>
   496→            </GlassCard>
   497→          </motion.div>
   498→        )}
   499→      </AnimatePresence>
   500→
   501→      {/* Add Lesson Form */}
   502→      <AnimatePresence>
   503→        {addMode === 'lesson' && (
   504→          <motion.div {...collapseExpand}>
   505→            <GlassCard className="p-5" glow>
   506→              <div className="flex items-center justify-between mb-4">
   507→                <h3 className="text-base font-bold gradient-text">New Lesson</h3>
   508→                <button onClick={() => { setAddMode(null); setShowVideoSearch(false); setSelectedExistingVideo(null); }} className="w-8 h-8 rounded-lg bg-muted/50 flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors">
   509→                  <X className="w-4 h-4" />
   510→                </button>
   511→              </div>
   512→              <div className="space-y-3">
