     1→'use client';
     2→
     3→import { useState, useMemo } from 'react';
     4→import { motion, AnimatePresence } from 'framer-motion';
     5→import {
     6→  Plus, Trash2, BookOpen, FolderOpen,
     7→  Video, Radio, FileText, StickyNote, ChevronDown,
     8→  Link as LinkIcon, Edit3, Eye, X, Save,
     9→  MessageSquare, Send, Users,
    10→} from 'lucide-react';
    11→import { toast } from 'sonner';
    12→import { GlassCard } from '@/components/shared/GlassCard';
    13→import { GradientButton } from '@/components/shared/GradientButton';
    14→import { LoadingSkeleton } from '@/components/shared/LoadingSkeleton';
    15→import { EmptyState } from '@/components/shared/EmptyState';
    16→import {
    17→  useCourseSubjects, useCurriculum, useRemoveSubjectFromCourse, useCourse,
    18→  useCreateChapter, useUpdateChapter, useDeleteChapter,
    19→  useCreateLesson, useUpdateLesson, useDeleteLesson,
    20→  useReviews, useReplyReview,
    21→} from '@/lib/api-hooks';
    22→import { useNavigationStore } from '@/lib/store';
    23→import { cn, formatDurationShort, relativeTime } from '@/lib/utils';
    24→import { staggerChildren, collapseExpand } from '@/lib/animations';
    25→
    26→type AddMode = 'chapter' | 'lesson' | null;
    27→type EditMode = 'chapter' | 'lesson' | null;
    28→
    29→const lessonTypeIcon: Record<string, typeof Video> = {
    30→  video: Video, link: LinkIcon, youtube: Video, document: FileText,
    31→};
    32→const lessonTypeLabel: Record<string, string> = {
    33→  video: 'Video', link: 'Link', youtube: 'YouTube', document: 'Document/PDF',
    34→};
    35→const lessonTypeColor: Record<string, string> = {
    36→  video: 'bg-neutral-100 text-neutral-700 dark:bg-neutral-800/30 dark:text-neutral-300',
    37→  link: 'bg-blue-100 text-blue-700 dark:bg-blue-900/30 dark:text-blue-400',
    38→  youtube: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
    39→  document: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
    40→};
    41→
    42→export function CourseCurriculum() {
    43→  const params = useNavigationStore((s) => s.params);
    44→  const navigate = useNavigationStore((s) => s.navigate);
    45→  const courseId = params.courseId || '';
    46→
    47→  const { data: courseData, loading: courseLoading, error: courseError } = useCourse(courseId);
    48→  const { data: subjectsData, loading: subjectsLoading, refetch: refetchSubjects } = useCourseSubjects(courseId);
    49→  const { data: curriculumData, loading: curriculumLoading, refetch: refetchCurriculum } = useCurriculum(courseId);
    50→  const { mutate: removeSubject } = useRemoveSubjectFromCourse(courseId);
    51→  const { mutate: createChapter, loading: creatingChapter } = useCreateChapter(courseId);
    52→  const { mutate: updateChapter, loading: updatingChapter } = useUpdateChapter(courseId);
    53→  const { mutate: deleteChapter, loading: deletingChapter } = useDeleteChapter(courseId);
    54→  const { mutate: createLesson, loading: creatingLesson } = useCreateLesson(courseId);
    55→  const { mutate: updateLesson, loading: updatingLesson } = useUpdateLesson(courseId);
    56→  const { mutate: deleteLesson, loading: deletingLesson } = useDeleteLesson(courseId);
    57→
    58→  // Reviews
    59→  const { data: reviewsData, refetch: refetchReviews } = useReviews(1);
    60→  const { mutate: replyReview, loading: replyingReview } = useReplyReview();
    61→
    62→  const course = courseData?.course || courseData;
    63→  const subjects = useMemo(() => subjectsData?.subjects || [], [subjectsData]);
    64→  const curriculum = curriculumData?.curriculum || { subjects: [], chapters: [], lessons: [], resources: [] };
    65→  const reviews = useMemo(() => reviewsData?.reviews || [], [reviewsData]);
    66→
    67→  // Expand/collapse state
    68→  const [expandedSubjects, setExpandedSubjects] = useState<Set<string>>(new Set());
    69→  const [expandedChapters, setExpandedChapters] = useState<Set<string>>(new Set());
    70→
    71→  // Inline add form state
    72→  const [addMode, setAddMode] = useState<AddMode>(null);
    73→  const [addSubjectId, setAddSubjectId] = useState<string>('');
    74→  const [addChapterId, setAddChapterId] = useState<string>('');
    75→
    76→  // Chapter add form
    77→  const [chapterTitle, setChapterTitle] = useState('');
    78→  const [chapterDesc, setChapterDesc] = useState('');
    79→
    80→  // Lesson add form
    81→  const [lessonTitle, setLessonTitle] = useState('');
    82→  const [lessonType, setLessonType] = useState<'video' | 'link' | 'youtube' | 'document'>('video');
    83→  const [lessonVideoUrl, setLessonVideoUrl] = useState('');
    84→  const [lessonDesc, setLessonDesc] = useState('');
    85→  const [lessonIsPreview, setLessonIsPreview] = useState(false);
    86→
    87→  // Edit state
    88→  const [editMode, setEditMode] = useState<EditMode>(null);
    89→  const [editId, setEditId] = useState<string>('');
    90→  const [editSubjectId, setEditSubjectId] = useState<string>('');
    91→
    92→  // Chapter edit form
    93→  const [editChapterTitle, setEditChapterTitle] = useState('');
    94→  const [editChapterDesc, setEditChapterDesc] = useState('');
    95→
    96→  // Lesson edit form
    97→  const [editLessonTitle, setEditLessonTitle] = useState('');
    98→  const [editLessonType, setEditLessonType] = useState<'video' | 'link' | 'youtube' | 'document'>('video');
    99→  const [editLessonVideoUrl, setEditLessonVideoUrl] = useState('');
   100→  const [editLessonDesc, setEditLessonDesc] = useState('');
   101→  const [editLessonIsPreview, setEditLessonIsPreview] = useState(false);
   102→
   103→  // Reply form
   104→  const [replyTexts, setReplyTexts] = useState<Record<number, string>>({});
   105→
   106→  // Removing subject
   107→  const [removingId, setRemovingId] = useState<string | number | null>(null);
   108→
   109→  // Group chapters and lessons by subject
   110→  const subjectDetails = useMemo(() => {
   111→    return subjects.map((cs: any) => {
   112→      const subjId = cs.subjectId || cs.subject_id;
   113→      const subjName = cs.subjectName || cs.subject_name || cs.name || 'Unknown Subject';
   114→      const subjSlug = cs.subjectSlug || cs.subject_slug || cs.slug || '';
   115→      const subjTechId = cs.technologyId || cs.technology_id || 0;
   116→      const subjSemester = cs.semester || '';
   117→
   118→      const subjectChapters = (curriculum.chapters || []).filter(
   119→        (ch: any) => String(ch.subjectId || ch.subject_id) === String(subjId)
   120→      ).sort((a: any, b: any) => (a.sortOrder || a.sort_order || 0) - (b.sortOrder || b.sort_order || 0));
   121→
   122→      const chapterIds = subjectChapters.map((ch: any) => ch.id);
   123→      const subjectLessons = (curriculum.lessons || []).filter(
   124→        (l: any) => chapterIds.includes(l.chapterId || l.chapter_id) || String(l.subjectId || l.subject_id) === String(subjId)
   125→      );
   126→
   127→      const videoCount = subjectLessons.filter((l: any) => (l.lessonType || l.lesson_type) === 'video').length;
   128→      const docCount = subjectLessons.filter((l: any) => (l.lessonType || l.lesson_type) === 'document').length;
   129→      const liveCount = subjectLessons.filter((l: any) => (l.lessonType || l.lesson_type) === 'link').length;
   130→
   131→      return {
   132→        ...cs,
   133→        subjectId: subjId,
   134→        name: subjName,
   135→        slug: subjSlug,
   136→        technologyId: subjTechId,
   137→        semester: subjSemester,
   138→        chapters: subjectChapters,
   139→        lessons: subjectLessons,
   140→        chapterCount: subjectChapters.length,
   141→        lessonCount: subjectLessons.length,
   142→        videoCount,
   143→        docCount,
   144→        liveCount,
   145→      };
   146→    });
   147→  }, [subjects, curriculum]);
   148→
   149→  const toggleSubject = (subjId: string) => {
   150→    setExpandedSubjects((prev) => {
   151→      const next = new Set(prev);
   152→      if (next.has(String(subjId))) next.delete(String(subjId));
   153→      else next.add(String(subjId));
   154→      return next;
   155→    });
   156→  };
   157→
   158→  const toggleChapter = (chapterId: string) => {
   159→    setExpandedChapters((prev) => {
   160→      const next = new Set(prev);
   161→      if (next.has(chapterId)) next.delete(chapterId);
   162→      else next.add(chapterId);
   163→      return next;
   164→    });
   165→  };
   166→
   167→  const startAddChapter = (subjectId: string) => {
   168→    setAddMode('chapter');
   169→    setAddSubjectId(String(subjectId));
   170→    setAddChapterId('');
   171→    setChapterTitle('');
   172→    setChapterDesc('');
   173→    // Close any edit form
   174→    setEditMode(null);
   175→    setEditId('');
   176→  };
   177→
   178→  const startAddLesson = (subjectId: string, chapterId: string) => {
   179→    setAddMode('lesson');
   180→    setAddSubjectId(String(subjectId));
   181→    setAddChapterId(chapterId);
   182→    setLessonTitle('');
   183→    setLessonVideoUrl('');
   184→    setLessonDesc('');
   185→    setLessonIsPreview(false);
   186→    setLessonType('video');
   187→    // Close any edit form
   188→    setEditMode(null);
   189→    setEditId('');
   190→  };
   191→
   192→  const startEditChapter = (chapter: any, subjectId: string) => {
   193→    setEditMode('chapter');
   194→    setEditId(chapter.id);
   195→    setEditSubjectId(String(subjectId));
   196→    setEditChapterTitle(chapter.title || '');
   197→    setEditChapterDesc(chapter.description || '');
   198→    // Close any add form
   199→    setAddMode(null);
   200→  };
   201→
   202→  const startEditLesson = (lesson: any, subjectId: string) => {
   203→    setEditMode('lesson');
   204→    setEditId(lesson.id);
   205→    setEditSubjectId(String(subjectId));
   206→    setEditLessonTitle(lesson.title || '');
   207→    setEditLessonType((lesson.lessonType || lesson.lesson_type || 'video') as any);
   208→    setEditLessonVideoUrl(lesson.videoUrl || lesson.video_url || '');
   209→    setEditLessonDesc(lesson.description || '');
   210→    setEditLessonIsPreview(!!lesson.isPreview);
   211→    // Close any add form
   212→    setAddMode(null);
   213→  };
   214→
   215→  const cancelEdit = () => {
   216→    setEditMode(null);
   217→    setEditId('');
   218→  };
   219→
   220→  const handleCreateChapter = async () => {
   221→    if (!chapterTitle.trim()) { toast.error('Enter chapter title'); return; }
   222→    try {
   223→      const subject = subjectDetails.find((s: any) => String(s.subjectId) === addSubjectId);
   224→      const result = await createChapter({
   225→        title: chapterTitle.trim(),
   226→        subjectId: addSubjectId,
   227→        description: chapterDesc.trim() || undefined,
   228→        sortOrder: (subject?.chapters?.length || 0) + 1,
   229→      });
   230→      if (result) {
   231→        toast.success('Chapter created');
   232→        setAddMode(null);
   233→        refetchCurriculum();
   234→      } else {
   235→        toast.error('Failed to create chapter');
   236→      }
   237→    } catch (err: any) {
   238→      toast.error(err?.message || 'Failed to create chapter');
   239→    }
   240→  };
   241→
   242→  const handleUpdateChapter = async () => {
   243→    if (!editChapterTitle.trim()) { toast.error('Enter chapter title'); return; }
   244→    try {
   245→      const result = await updateChapter({
   246→        chapterId: editId,
   247→        title: editChapterTitle.trim(),
   248→        description: editChapterDesc.trim() || undefined,
   249→      });
   250→      if (result) {
   251→        toast.success('Chapter updated');
   252→        setEditMode(null);
   253→        setEditId('');
   254→        refetchCurriculum();
   255→      } else {
   256→        toast.error('Failed to update chapter');
   257→      }
   258→    } catch (err: any) {
   259→      toast.error(err?.message || 'Failed to update chapter');
   260→    }
   261→  };
   262→
   263→  const handleDeleteChapter = async (chapterId: string) => {
   264→    if (!confirm('Delete this chapter and all its lessons?')) return;
   265→    try {
   266→      const result = await deleteChapter({ chapterId });
   267→      if (result !== null) {
   268→        toast.success('Chapter deleted');
   269→        refetchCurriculum();
   270→      } else {
   271→        toast.error('Failed to delete chapter');
   272→      }
   273→    } catch (err: any) {
   274→      toast.error(err?.message || 'Failed to delete chapter');
   275→    }
   276→  };
   277→
   278→  const handleCreateLesson = async () => {
   279→    if (!lessonTitle.trim()) { toast.error('Enter lesson title'); return; }
   280→    if (!addChapterId) { toast.error('Select a chapter'); return; }
   281→    try {
   282→      const subject = subjectDetails.find((s: any) => String(s.subjectId) === addSubjectId);
   283→      const chapterLessons = (subject?.lessons || []).filter((l: any) => (l.chapterId || l.chapter_id) === addChapterId);
   284→      const result = await createLesson({
   285→        title: lessonTitle.trim(),
   286→        chapterId: addChapterId,
   287→        subjectId: addSubjectId as any,
   288→        lessonType,
   289→        videoUrl: lessonVideoUrl.trim() || undefined,
   290→        description: lessonDesc.trim() || undefined,
   291→        isPreview: lessonIsPreview,
   292→        sortOrder: chapterLessons.length + 1,
   293→      });
   294→      if (result) {
   295→        toast.success('Lesson added');
   296→        setAddMode(null);
   297→        refetchCurriculum();
   298→      } else {
   299→        toast.error('Failed to add lesson');
   300→      }
   301→    } catch (err: any) {
   302→      toast.error(err?.message || 'Failed to add lesson');
   303→    }
   304→  };
   305→
   306→  const handleUpdateLesson = async () => {
   307→    if (!editLessonTitle.trim()) { toast.error('Enter lesson title'); return; }
   308→    try {
   309→      const result = await updateLesson({
   310→        lessonId: editId,
   311→        title: editLessonTitle.trim(),
   312→        description: editLessonDesc.trim() || undefined,
   313→        lessonType: editLessonType,
   314→        videoUrl: editLessonVideoUrl.trim() || undefined,
   315→        isPreview: editLessonIsPreview,
   316→      });
   317→      if (result) {
   318→        toast.success('Lesson updated');
   319→        setEditMode(null);
   320→        setEditId('');
   321→        refetchCurriculum();
   322→      } else {
   323→        toast.error('Failed to update lesson');
   324→      }
   325→    } catch (err: any) {
   326→      toast.error(err?.message || 'Failed to update lesson');
   327→    }
   328→  };
   329→
   330→  const handleDeleteLesson = async (lessonId: string) => {
   331→    if (!confirm('Delete this lesson?')) return;
   332→    try {
   333→      const result = await deleteLesson({ lessonId });
   334→      if (result !== null) {
   335→        toast.success('Lesson deleted');
   336→        refetchCurriculum();
   337→      } else {
   338→        toast.error('Failed to delete lesson');
   339→      }
   340→    } catch (err: any) {
   341→      toast.error(err?.message || 'Failed to delete lesson');
   342→    }
   343→  };
   344→
   345→  const handleRemoveSubject = async (subjectId: string | number) => {
   346→    if (!confirm('Remove this subject from the course? This will not delete the subject itself.')) return;
   347→    setRemovingId(subjectId);
   348→    try {
   349→      const result = await removeSubject({ subjectId: typeof subjectId === 'string' ? subjectId : Number(subjectId) });
   350→      if (result !== null) {
   351→        toast.success('Subject removed');
   352→        refetchSubjects();
   353→        refetchCurriculum();
   354→      } else {
   355→        toast.error('Failed to remove subject');
   356→      }
   357→    } catch (err: any) {
   358→      toast.error(err?.message || 'Failed to remove subject');
   359→    }
   360→    setRemovingId(null);
   361→  };
   362→
   363→  const handleReply = async (reviewId: number) => {
   364→    const text = replyTexts[reviewId]?.trim();
   365→    if (!text) return;
   366→    try {
   367→      const result = await replyReview({ reviewId, replyText: text });
   368→      if (result) {
   369→        toast.success('Reply posted');
   370→        setReplyTexts((prev) => ({ ...prev, [reviewId]: '' }));
   371→        refetchReviews();
   372→      } else {
   373→        toast.error('Failed to post reply');
   374→      }
   375→    } catch (err: any) {
   376→      toast.error(err?.message || 'Failed to post reply');
   377→    }
   378→  };
   379→
   380→  // Course sub-navigation tabs
   381→  const courseTabs = [
   382→    { id: 'overview', label: 'Overview', icon: BookOpen, page: 'course-detail' as const },
   383→    { id: 'curriculum', label: 'Curriculum', icon: FolderOpen, page: 'course-curriculum' as const },
   384→    { id: 'videos', label: 'Videos', icon: Video, page: 'course-videos' as const },
   385→    { id: 'resources', label: 'Resources', icon: FileText, page: 'course-resources' as const },
   386→    { id: 'instructors', label: 'Instructors', icon: Users, page: 'course-instructors' as const },
   387→    { id: 'settings', label: 'Settings', icon: Radio, page: 'course-settings' as const },
   388→    { id: 'live', label: 'Live', icon: Radio, page: 'course-live' as const },
   389→  ];
   390→
   391→  if (courseLoading || subjectsLoading || curriculumLoading) {
   392→    return <LoadingSkeleton type="card" count={3} />;
   393→  }
   394→
   395→  if (courseError) {
   396→    return (
   397→      <div className="flex flex-col items-center justify-center py-20 space-y-4">
   398→        <p className="text-sm text-red-500">Failed to load course: {courseError}</p>
   399→        <GradientButton size="sm" onClick={() => refetchCurriculum()}>Retry</GradientButton>
   400→      </div>
   401→    );
   402→  }
   403→
   404→  return (
   405→    <div className="space-y-6">
   406→      {/* Header */}
   407→      <motion.div initial={{ opacity: 0, y: 8 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }}>
   408→        <div className="flex items-center justify-between gap-4">
   409→          <div>
   410→            <h1 className="text-2xl md:text-3xl font-extrabold text-foreground">
   411→              <span className="gradient-text">Curriculum</span>
   412→            </h1>
   413→            <p className="text-muted-foreground mt-1 text-sm">{course?.title || ''}</p>
   414→          </div>
   415→          <GradientButton
   416→            size="sm"
   417→            icon={<Plus className="w-4 h-4" />}
   418→            onClick={() => navigate('course-add-subject', { courseId })}
   419→          >
   420→            Add Subject
   421→          </GradientButton>
   422→        </div>
   423→      </motion.div>
   424→
   425→      {/* Course Sub-Navigation */}
   426→      <motion.div
   427→        initial={{ opacity: 0, y: 8 }}
   428→        animate={{ opacity: 1, y: 0 }}
   429→        transition={{ delay: 0.1, duration: 0.3 }}
   430→      >
   431→        <div className="flex items-center gap-1 overflow-x-auto pb-1">
   432→          {courseTabs.map((tab) => (
   433→            <button
   434→              key={tab.id}
   435→              onClick={() => navigate(tab.page, { courseId })}
   436→              className={cn(
   437→                'relative flex items-center gap-2 px-4 py-2.5 text-sm font-semibold rounded-xl transition-colors whitespace-nowrap',
   438→                tab.id === 'curriculum'
   439→                  ? 'bg-black/[0.05] dark:bg-white/[0.06] text-foreground'
   440→                  : 'text-muted-foreground hover:text-foreground hover:bg-black/[0.02] dark:hover:bg-white/[0.02]'
   441→              )}
   442→            >
   443→              <tab.icon className="w-4 h-4" />
   444→              {tab.label}
   445→              {tab.id === 'curriculum' && (
   446→                <motion.div
   447→                  className="absolute bottom-0 left-2 right-2 h-[2px] rounded-full bg-black dark:bg-white"
   448→                  layoutId="curriculum-tab"
   449→                  transition={{ type: 'spring', stiffness: 350, damping: 30 }}
   450→                />
   451→              )}
   452→            </button>
   453→          ))}
   454→        </div>
   455→      </motion.div>
   456→
   457→      {/* Subjects List */}
   458→      {subjectDetails.length === 0 ? (
   459→        <EmptyState
   460→          icon={FolderOpen}
   461→          title="No subjects added yet"
   462→          description="Add subjects to organize your course content. Each subject can have chapters, lessons, videos, and more."
   463→          actionLabel="Add Subject"
   464→          onAction={() => navigate('course-add-subject', { courseId })}
   465→        />
   466→      ) : (
   467→        <div className="space-y-3">
   468→          {subjectDetails.map((subject: any, i: number) => {
   469→            const subjKey = String(subject.subjectId || subject.id);
   470→            const isSubjectExpanded = expandedSubjects.has(subjKey);
   471→
   472→            return (
   473→              <motion.div
   474→                key={subjKey || i}
   475→                initial={{ opacity: 0, y: 12 }}
   476→                animate={{ opacity: 1, y: 0 }}
   477→                transition={{ delay: i * 0.05, duration: 0.3 }}
   478→              >
   479→                <GlassCard className="p-0 overflow-hidden">
   480→                  {/* Subject Header - Clickable to expand */}
   481→                  <div
   482→                    className="p-4 md:p-5 cursor-pointer hover:bg-black/[0.01] dark:hover:bg-white/[0.01] transition-colors"
   483→                    onClick={() => toggleSubject(subjKey)}
   484→                  >
   485→                    <div className="flex items-start gap-3 md:gap-4">
   486→                      {/* Chevron Indicator */}
   487→                      <motion.div
   488→                        animate={{ rotate: isSubjectExpanded ? 0 : -90 }}
   489→                        transition={{ duration: 0.2 }}
   490→                        className="flex-shrink-0 mt-1"
   491→                      >
   492→                        <ChevronDown className="w-4 h-4 text-muted-foreground" />
   493→                      </motion.div>
   494→
   495→                      {/* Subject Icon */}
   496→                      <div className="w-10 h-10 md:w-12 md:h-12 rounded-2xl bg-gradient-to-br from-neutral-900 via-neutral-800 to-black dark:from-white dark:via-neutral-200 dark:to-white flex items-center justify-center shadow-[0_4px_15px_rgba(0,0,0,0.15)] flex-shrink-0">
   497→                        <BookOpen className="w-4 h-4 md:w-5 md:h-5 text-white dark:text-neutral-900" />
   498→                      </div>
   499→
   500→                      {/* Subject Info */}
   501→                      <div className="flex-1 min-w-0">
   502→                        <div className="flex items-center gap-2">
   503→                          <h3 className="text-base font-bold text-foreground truncate">{subject.name}</h3>
   504→                          {subject.semester && (
   505→                            <span className="text-[10px] font-bold px-2 py-0.5 rounded-full bg-neutral-100 text-neutral-700 dark:bg-neutral-800/30 dark:text-neutral-300">
   506→                              {subject.semester}
   507→                            </span>
   508→                          )}
   509→                        </div>
   510→
   511→                        {/* Stats row */}
