   174→/** Convert a page name to a URL path, including dynamic params */
   175→export function pageToUrl(page: string, params?: Record<string, unknown>): string {
   176→  let path = pageToPath[page] || `/${page}`;
   177→
   178→  // Append dynamic params as URL segments
   179→  if (params) {
   180→    const extraSegments: string[] = [];
   181→    if (params.courseId) extraSegments.push(String(params.courseId));
   182→    // Course detail deep links: /course/detail/{courseId}/{tab}
   183→    if (page === 'course-detail' && params.tab && params.courseId) {
   184→      extraSegments.push(String(params.tab));
   185→    }
   186→    if (params.videoId) extraSegments.push(String(params.videoId));
   187→    if (params.instructorId) extraSegments.push(String(params.instructorId));
   188→    if (params.query) extraSegments.push(encodeURIComponent(String(params.query)));
   189→    if (params.userId && params.notificationId) {
   190→      // Notification detail: /notifications/{userId}/{notificationId}[/slug]
   191→      extraSegments.push(String(params.userId));
   192→      extraSegments.push(String(params.notificationId));
   193→      if (params.notificationSlug) extraSegments.push(String(params.notificationSlug));
   194→    } else if (params.userId) {
   195→      extraSegments.push(String(params.userId));
   196→    }
   197→    if (extraSegments.length > 0) {
   198→      path += '/' + extraSegments.join('/');
   199→    }
   200→  }
   201→
   202→  return path;
   203→}

   205→/** Convert a URL path back to a page name and params */
   206→export function urlToPage(urlPath: string): { page: string; params: Record<string, string> } {
   207→  // Normalize: remove trailing slash
   208→  const normalized = urlPath.replace(/\/+$/, '') || '/';
   209→
   210→  // Try exact match first
   211→  const exactMatch = pathToPage[normalized];
   212→  if (exactMatch) {
   213→    return { page: exactMatch, params: {} };
   214→  }
   215→
   216→  // Try matching with dynamic segments
   217→  // e.g., /course/detail/c1 → { page: 'course-detail', params: { courseId: 'c1' } }
   218→  // e.g., /search/web%20dev → { page: 'search', params: { query: 'web dev' } }
   219→  // e.g., /instructor/profile/ins1 → { page: 'instructor-profile', params: { instructorId: 'ins1' } }
   220→
   221→  const segments = normalized.split('/').filter(Boolean);
   222→
   223→  // Try progressively shorter paths
   224→  for (let i = segments.length; i >= 1; i--) {
   225→    const prefix = '/' + segments.slice(0, i).join('/');
   226→    const match = pathToPage[prefix];
   227→    if (match) {
   228→      const extraSegments = segments.slice(i);
   229→      const params: Record<string, string> = {};
   230→
   231→      // Assign extra segments to known param keys based on page type
   232→      if (match === 'course-detail' && extraSegments[0]) {
   233→        params.courseId = extraSegments[0];
   234→        // Support deep link tabs: /course/detail/{courseId}/{tab}
   235→        // tab can be: 'overview', 'curriculum', 'reviews', 'instructor'
   236→        if (extraSegments[1]) {
   237→          const validTabs = ['overview', 'curriculum', 'reviews', 'instructor'];
   238→          if (validTabs.includes(extraSegments[1])) {
   239→            params.tab = extraSegments[1];
   240→          }
   241→        }
   242→      }
   243→      else if (match === 'video-player' && extraSegments[0]) params.videoId = extraSegments[0];
   244→      else if (match === 'instructor-profile' && extraSegments[0]) params.instructorId = extraSegments[0];
   245→      else if (match === 'search' && extraSegments[0]) params.query = decodeURIComponent(extraSegments[0]);
   246→      else if (match === 'profile' && extraSegments[0]) params.userId = extraSegments[0];
   247→      else if (match === 'notifications' && extraSegments.length >= 2) {
   248→        // /notifications/{userId}/{notificationId}[/slug] → notification-detail page
   249→        return { page: 'notification-detail' as Page, params: { userId: extraSegments[0], notificationId: extraSegments[1], ...(extraSegments.length > 2 ? { notificationSlug: extraSegments.slice(2).join('/') } : {}) } };
   250→      }
   251→      else {
   252→        // Generic: store as param0, param1, etc.
   253→        extraSegments.forEach((seg, idx) => {
   254→          params[`param${idx}`] = seg;
   255→        });
   256→      }
   257→
   258→      return { page: match, params };
   259→    }
   260→  }
   261→
   262→  // No match found → 404
   263→  return { page: 'error-404', params: {} };
   264→}

   266→// ============ NAVIGATION STORE ============
   267→interface NavigationState {
   268→  currentPage: Page;
   269→  pageParams: Record<string, unknown>;
   270→  navigate: (page: Page, params?: Record<string, unknown>) => void;
   271→  goBack: () => void;
   272→  history: Array<{ page: Page; params: Record<string, unknown> }>;
   273→  sidebarOpen: boolean;
   274→  toggleSidebar: () => void;
   275→  setSidebarOpen: (open: boolean) => void;
   276→  syncFromUrl: (urlPath: string) => void;
   277→}
   278→
   279→let isNavigatingProgrammatically = false;
   280→
   281→export const useNavigationStore = create<NavigationState>((set, get) => ({
   282→  currentPage: 'home',
   283→  pageParams: {},
   284→  history: [],
   285→  sidebarOpen: false,
   286→  toggleSidebar: () => set((s) => ({ sidebarOpen: !s.sidebarOpen })),
   287→  setSidebarOpen: (open) => set({ sidebarOpen: open }),
   288→  navigate: (page, params = {}) => {
   289→    const state = get();
   290→    const url = pageToUrl(page, params);
   291→
   292→    // Update browser URL
   293→    if (typeof window !== 'undefined') {
   294→      isNavigatingProgrammatically = true;
   295→      window.history.pushState({ page, params }, '', url);
   296→      // Reset the flag after a microtask
   297→      queueMicrotask(() => { isNavigatingProgrammatically = false; });
   298→    }
   299→
   300→    set({
   301→      currentPage: page,
   302→      pageParams: params,
   303→      history: [...state.history, { page: state.currentPage, params: state.pageParams }],
   304→    });
   305→  },
   306→  goBack: () => {
   307→    const state = get();
   308→    if (typeof window !== 'undefined') {
   309→      window.history.back();
   310→    }
   311→    if (state.history.length > 0) {
   312→      const prev = state.history[state.history.length - 1];
   313→      set({
   314→        currentPage: prev.page,
   315→        pageParams: prev.params,
   316→        history: state.history.slice(0, -1),
   317→      });
   318→    }
   319→  },
   320→  syncFromUrl: (urlPath: string) => {
   321→    const { page, params } = urlToPage(urlPath);
   322→    if (page !== get().currentPage || JSON.stringify(params) !== JSON.stringify(get().pageParams)) {
   323→      set({
   324→        currentPage: page as Page,
   325→        pageParams: params,
   326→      });
   327→    }
   328→  },
   329→}));

   370→// ============ AUTH PERSISTENCE ============
   371→const AUTH_STORAGE_KEY = 'dakkho-auth-session';
   372→const PENDING_VERIFICATION_KEY = 'dakkho-pending-verification';
   373→
   374→const loadAuthSession = (): { user: User | null; isAuthenticated: boolean; needsVerification: boolean; pendingVerificationEmail: string | null } => {
   375→  if (typeof window === 'undefined') return { user: null, isAuthenticated: false, needsVerification: false, pendingVerificationEmail: null };
   376→  try {
   377→    // Check for pending verification state first
   378→    const pendingStored = localStorage.getItem(PENDING_VERIFICATION_KEY);
   379→    const pendingToken = getPendingVerificationToken();
   380→    if (pendingStored && pendingToken) {
   381→      const parsed = JSON.parse(pendingStored);
   382→      return { user: parsed.user, isAuthenticated: false, needsVerification: true, pendingVerificationEmail: parsed.user?.email || null };
   383→    }
   384→
   385→    const stored = localStorage.getItem(AUTH_STORAGE_KEY);
   386→    const token = localStorage.getItem('dakkho_student_token');
   387→    if (stored && token) {
   388→      const parsed = JSON.parse(stored);
   389→      if (parsed.expiresAt && Date.now() < parsed.expiresAt) {
   390→        return { user: parsed.user, isAuthenticated: parsed.isAuthenticated, needsVerification: false, pendingVerificationEmail: null };
   391→      }
   392→      localStorage.removeItem(AUTH_STORAGE_KEY);
   393→      localStorage.removeItem('dakkho_student_token');
   394→    }
   395→  } catch {}
   396→  return { user: null, isAuthenticated: false, needsVerification: false, pendingVerificationEmail: null };
   397→};
   398→
   399→const saveAuthSession = (user: User | null, isAuthenticated: boolean) => {
   400→  if (typeof window === 'undefined') return;
   401→  try {
   402→    if (isAuthenticated && user) {
   403→      const expiresAt = Date.now() + 30 * 24 * 60 * 60 * 1000;
   404→      localStorage.setItem(AUTH_STORAGE_KEY, JSON.stringify({ user, isAuthenticated, expiresAt }));
   405→    } else {
   406→      localStorage.removeItem(AUTH_STORAGE_KEY);
   407→    }
   408→  } catch {}
   409→};
   410→
   411→const savePendingVerification = (user: User, token: string) => {
   412→  if (typeof window === 'undefined') return;
   413→  try {
   414→    localStorage.setItem(PENDING_VERIFICATION_KEY, JSON.stringify({ user }));
   415→    setPendingVerificationToken(token);
   416→  } catch {}
   417→};
   418→
   419→const clearPendingVerification = () => {
   420→  if (typeof window === 'undefined') return;
   421→  try {
   422→    localStorage.removeItem(PENDING_VERIFICATION_KEY);
   423→    clearPendingVerificationToken();
   424→  } catch {}
   425→};

   486→        const user: User = {
   487→          id: res.userId || '',
   488→          fullName: data.fullName,
   489→          email: data.email,
   490→          institute: res.user?.instituteName || undefined,
   491→          instituteId: data.instituteId,
   492→          technology: data.technology,
   493→          technologyName: res.user?.technologyName || undefined,
   494→          avatarUrl: '',
   495→          role: 'student',
   496→          isNewUser: true,
   497→          emailVerified: false,
   498→          enrolledCourseIds: [],
   499→        };
   500→        // Do NOT set isAuthenticated — user must verify OTP first.
   501→        // Store the token in the pending key so API calls (like verify-otp) work,

   503→        savePendingVerification(user, res.token);

   506→      } else {
   507→        throw new Error(res.message || 'Signup failed');
   508→      }
   509→    } catch (err: any) {
   510→      set({ isLoading: false });
   511→      throw new Error(err.message || 'Signup failed. Please try again.');
   512→    }

   640→  toggleTheme: () => void;
   641→  loadFromPreferences: (mode: 'light' | 'dark' | 'system') => void;

   643→
   644→const applyThemeMode = (mode: 'light' | 'dark' | 'system'): 'light' | 'dark' => {
   645→  let effectiveTheme: 'light' | 'dark';
   646→  if (mode === 'system') {
   647→    effectiveTheme = window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light';
   648→  } else {
   649→    effectiveTheme = mode;
   650→  }
   651→  if (effectiveTheme === 'dark') {
   652→    document.documentElement.classList.add('dark');
   653→  } else {
   654→    document.documentElement.classList.remove('dark');
   655→  }
   656→  return effectiveTheme;
   657→};

   664→      localStorage.setItem('dakkho_theme_mode', mode);
   665→      const effectiveTheme = applyThemeMode(mode);
   666→      set({ themeMode: mode, theme: effectiveTheme });
   667→    }
   668→  },
   669→  toggleTheme: () => {
   670→    const next = get().theme === 'light' ? 'dark' : 'light';

   674→    }

   676→      document.documentElement.classList.add('dark');
   677→    } else {
   678→      document.documentElement.classList.remove('dark');
   679→    }
   680→  },
   681→  loadFromPreferences: (mode) => {
   682→    if (typeof window !== 'undefined') {
   683→      localStorage.setItem('dakkho_theme_mode', mode);
   684→      const effectiveTheme = applyThemeMode(mode);
   685→      set({ themeMode: mode, theme: effectiveTheme });
   686→    }

   820→        n.id === id ? { ...n, isRead: true } : n

   827→    set((s) => {
   828→      const updated = s.notifications.map((n) => ({ ...n, isRead: true }));

   830→      return { notifications: updated };

   834→  addNotification: (notification) => {

   838→        return s;
   839→      }

   841→      saveNotifToStorage(updated);

   848→      const unique = newNotifs.filter((n) => !existingIds.has(n.id));
   849→      if (unique.length === 0) return s;
   850→      const updated = [...unique, ...s.notifications];
   851→      saveNotifToStorage(updated);
   852→      return { notifications: updated };
   853→    });
   854→  },

   856→    set((s) => {
   857→      const updated = s.notifications.filter((n) => n.id !== id);
   858→      saveNotifToStorage(updated);
   859→      return { notifications: updated };
   860→    });
   861→  },
   862→  hydrateFromStorage: () => {
   863→    const stored = loadNotifFromStorage();
   864→    if (stored.length > 0) {
   865→      set({ notifications: stored });
   866→    }
   867→  },
