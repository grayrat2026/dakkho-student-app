    28→// Video duration detector
    29→function getVideoDuration(file: File): Promise<number> {
    30→  return new Promise((resolve) => {
    31→    const video = document.createElement('video');
    32→    video.preload = 'metadata';
    33→    video.onloadedmetadata = () => {
    34→      resolve(Math.round(video.duration));
    35→      URL.revokeObjectURL(video.src);
    36→    };
    37→    video.onerror = () => resolve(0);
    38→    video.src = URL.createObjectURL(file);
    39→  });
    40→}

   118→    return map;

   122→  const getLessonsForChapter = useCallback((chapterId: string) => {
   123→    const lessons = curriculum.lessons || [];

   125→      .filter((l: any) => (l.chapterId || l.chapter_id) === chapterId)
   126→      .sort((a: any, b: any) => (a.sortOrder || a.sort_order || 0) - (b.sortOrder || b.sort_order || 0));

   132→  const getSubjectSemester = (subject: any) => subject.semester || subject.semesterName || '';
   133→  const getSubjectTechId = (subject: any) => subject.technologyId || subject.technology_id;
   134→
   135→  // Available semesters from subjects matching selected technologies
   136→  const availableSemesters = useMemo(() => {
   137→    const semSet = new Set<string>();
   138→    allAvailableSubjects.forEach((s: any) => {
   139→      const techId = getSubjectTechId(s);
   140→      if (selectedTechs.size === 0 || selectedTechs.has(techId)) {
   141→        const sem = getSubjectSemester(s);
   142→        if (sem) semSet.add(sem);
   143→      }
   144→    });
   145→    return Array.from(semSet).sort();
   146→  }, [allAvailableSubjects, selectedTechs]);
   147→

   152→
   153→    return allAvailableSubjects.filter((s: any) => {
   154→      const techId = getSubjectTechId(s);
   155→      const sem = getSubjectSemester(s);
   156→      const id = getSubjectId(s);
   157→
   158→      // Filter by technology
   159→      if (selectedTechs.size > 0 && !selectedTechs.has(techId)) return false;
   160→      // Filter by semester
   161→      if (selectedSemesters.size > 0 && sem && !selectedSemesters.has(sem)) return false;
   162→      // Search query
   163→      if (subjectSearchQuery) {
   164→        const name = getSubjectName(s).toLowerCase();
   165→        if (!name.includes(subjectSearchQuery.toLowerCase())) return false;
   166→      }
   167→      // Exclude already added
   168→      if (addedIds.has(id)) return false;
   169→
   170→      return true;
   171→    });
   172→  }, [allAvailableSubjects, selectedTechs, selectedSemesters, subjectSearchQuery, subjects]);

   287→    let durationSec = 0;
   288→    if (durationParts.length === 2) {
   289→      durationSec = parseInt(durationParts[0]) * 60 + parseInt(durationParts[1]);
   290→    } else {
   291→      durationSec = parseInt(newLessonDuration) || 0;
   292→    }
   293→
   294→    const result = await createLesson({
   295→      title: newLessonTitle.trim(),
   296→      chapterId,
   297→      subjectId,
   298→      lessonType: newLessonType,
   299→      duration: durationSec || undefined,
   300→      isPreview: newLessonIsPreview,
   301→    });
   302→    if (result) {
   303→      toast.success('Lesson created');
   304→      resetLessonForm();
   305→      refetchCurriculum();
   306→    } else {
   307→      toast.error('Failed to create lesson');
   308→    }
   309→  };
   310→
   311→  const handleDeleteLesson = async (lessonId: string) => {
   312→    const result = await deleteLesson({ lessonId });
   313→    if (result !== null) {
   314→      toast.success('Lesson deleted');
   315→      refetchCurriculum();
   316→    }
   317→  };
   318→

   320→    const file = e.target.files?.[0];
   321→    if (!file) return;
   322→    const fd = new FormData();

   324→    fd.append('title', file.name);

   327→      toast.success('Resource uploaded');

   333→
   334→  const handleDeleteResource = async (resourceId: string) => {
   335→    const { apiDelete } = await import('@/lib/api-client');
   336→    try {
   337→      await apiDelete(`/courses/${courseId}/resources/${resourceId}`);
   338→      toast.success('Resource deleted');
   339→      refetchResources();

   341→      toast.error('Failed to delete resource');
   342→    }
   343→  };
   344→
   345→  const resetLessonForm = () => {

   347→    setNewLessonType('video');

   353→    setShowNewLesson(null);

   379→      {/* Course Subjects & Chapters */}

   397→                    <div className="flex items-center gap-3">

   412→                        className="w-8 h-8 rounded-lg flex items-center justify-center text-muted-foreground hover:text-red-500 hover:bg-red-50 dark:hover:bg-red-900/20 transition-colors"

   414→                        <Trash2 className="w-3.5 h-3.5" />

   416→                    </div>

   418→

   424→                        animate={{ height: 'auto', opacity: 1 }}

   436→                          />

   438→                            type="text"
   439→                            value={newChapterDesc}
   440→                            onChange={(e) => setNewChapterDesc(e.target.value)}
   441→                            placeholder="Chapter description (optional)"
   442→                            className="w-full px-3 py-2 rounded-lg text-sm premium-input placeholder:text-muted-foreground/60"
   443→                          />
   444→                          <div className="flex gap-2">

   450→                        </div>
   451→                      </motion.div>
   452→                    )}
   453→                  </AnimatePresence>
   454→
   455→                  {/* Chapters */}
   456→                  <div className="divide-y divide-black/[0.03] dark:divide-white/[0.03]">

   460→                        <ChapterSection
   461→                          key={chapter.id}
   462→                          chapter={chapter}
   463→                          lessons={chapterLessons}
   464→                          courseId={courseId}
   465→                          onDeleteChapter={handleDeleteChapter}
   466→                          onUpdateChapter={handleUpdateChapter}
   467→                          onDeleteLesson={handleDeleteLesson}
   468→                          onCreateLesson={handleCreateLesson}

   470→                          setEditingChapter={setEditingChapter}
   471→                          editChapterTitle={editChapterTitle}
   472→                          setEditChapterTitle={setEditChapterTitle}
   473→                          showNewLesson={showNewLesson}

   475→                          newLessonTitle={newLessonTitle}
   476→                          setNewLessonTitle={setNewLessonTitle}

   478→                          setNewLessonType={setNewLessonType}
   479→                          newLessonDuration={newLessonDuration}
   480→                          setNewLessonDuration={setNewLessonDuration}

   487→                          ccFile={ccFile}
   488→                          setCcFile={setCcFile}
   489→                          creatingLesson={creatingLesson}
   490→                          uploadingVideo={uploadingVideo}
