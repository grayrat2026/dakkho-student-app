     1→'use client';
     2→
     3→import { useState, useMemo } from 'react';
     4→import { motion, AnimatePresence } from 'framer-motion';
     5→import {
     6→  ArrowLeft, Plus, ChevronDown, ChevronRight, Trash2, Edit3,
     7→  FolderOpen, Video, FileText, Link as LinkIcon, Radio,
     8→  BookOpen, Play, Eye, X, Save, MessageSquare, Send,
     9→} from 'lucide-react';
    10→import { toast } from 'sonner';
    11→import { GlassCard } from '@/components/shared/GlassCard';
    12→import { GradientButton } from '@/components/shared/GradientButton';
    13→import { LoadingSkeleton } from '@/components/shared/LoadingSkeleton';
    14→import { EmptyState } from '@/components/shared/EmptyState';
    15→import {
    16→  useCurriculum, useCourseSubjects, useCourse,
    17→  useCreateChapter, useUpdateChapter, useDeleteChapter,
    18→  useCreateLesson, useUpdateLesson, useDeleteLesson,
    19→  useReviews, useReplyReview,
    20→} from '@/lib/api-hooks';
    21→import { useNavigationStore } from '@/lib/store';
    22→import { cn, formatDurationShort, relativeTime } from '@/lib/utils';
    23→import { staggerChildren, collapseExpand } from '@/lib/animations';
    24→
    25→type AddMode = 'chapter' | 'lesson' | null;
    26→type EditMode = 'chapter' | 'lesson' | null;
    27→
    28→export function CourseSubject() {
    29→  const params = useNavigationStore((s) => s.params);
    30→  const navigate = useNavigationStore((s) => s.navigate);
    31→  const goBack = useNavigationStore((s) => s.goBack);
    32→  const courseId = params.courseId || '';
    33→  const subjectId = params.subjectId || '';  // Can be UUID string or number
    34→
    35→  const { data: courseData, loading: courseLoading, error: courseError } = useCourse(courseId);
    36→  const { data: subjectsData, loading: subjectsLoading } = useCourseSubjects(courseId);
    37→  const { data: curriculumData, loading: curriculumLoading, refetch: refetchCurriculum } = useCurriculum(courseId);
    38→  const { mutate: createChapter, loading: creatingChapter } = useCreateChapter(courseId);
    39→  const { mutate: updateChapter, loading: updatingChapter } = useUpdateChapter(courseId);
    40→  const { mutate: deleteChapter, loading: deletingChapter } = useDeleteChapter(courseId);
    41→  const { mutate: createLesson, loading: creatingLesson } = useCreateLesson(courseId);
    42→  const { mutate: updateLesson, loading: updatingLesson } = useUpdateLesson(courseId);
    43→  const { mutate: deleteLesson, loading: deletingLesson } = useDeleteLesson(courseId);
    44→
    45→  // Reviews for this course
    46→  const { data: reviewsData, refetch: refetchReviews } = useReviews(1);
    47→  const { mutate: replyReview, loading: replyingReview } = useReplyReview();
    48→
    49→  const course = courseData?.course || courseData;
    50→  const allSubjects = subjectsData?.subjects || [];
    51→  const currentSubject = allSubjects.find((s: any) => String(s.subjectId || s.id) === String(subjectId));
    52→  const curriculum = curriculumData?.curriculum || { subjects: [], chapters: [], lessons: [], resources: [] };
    53→
    54→  // Filter data for this subject
    55→  const subjectChapters = useMemo(() => {
    56→    return (curriculum.chapters || [])
    57→      .filter((ch: any) => String(ch.subjectId || ch.subject_id) === String(subjectId))
    58→      .sort((a: any, b: any) => (a.sortOrder || a.sort_order || 0) - (b.sortOrder || b.sort_order || 0));
    59→  }, [curriculum, subjectId]);
    60→
    61→  const allLessons = useMemo(() => {
    62→    return (curriculum.lessons || [])
    63→      .filter((l: any) => String(l.subjectId || l.subject_id) === String(subjectId));
    64→  }, [curriculum, subjectId]);
    65→
    66→  const reviews = useMemo(() => reviewsData?.reviews || [], [reviewsData]);
    67→
    68→  // State
    69→  const [expandedChapters, setExpandedChapters] = useState<Set<string>>(new Set());
    70→  const [addMode, setAddMode] = useState<AddMode>(null);
    71→  const [selectedChapterId, setSelectedChapterId] = useState<string>('');
    72→
    73→  // Chapter add form
    74→  const [chapterTitle, setChapterTitle] = useState('');
    75→  const [chapterDesc, setChapterDesc] = useState('');
    76→
    77→  // Lesson add form
    78→  const [lessonTitle, setLessonTitle] = useState('');
    79→  const [lessonType, setLessonType] = useState<'video' | 'link' | 'youtube' | 'document'>('video');
    80→  const [lessonVideoUrl, setLessonVideoUrl] = useState('');
    81→  const [lessonDesc, setLessonDesc] = useState('');
    82→  const [lessonIsPreview, setLessonIsPreview] = useState(false);
    83→
    84→  // Edit state
    85→  const [editMode, setEditMode] = useState<EditMode>(null);
    86→  const [editId, setEditId] = useState<string>('');
    87→
    88→  // Chapter edit form
    89→  const [editChapterTitle, setEditChapterTitle] = useState('');
    90→  const [editChapterDesc, setEditChapterDesc] = useState('');
    91→
    92→  // Lesson edit form
    93→  const [editLessonTitle, setEditLessonTitle] = useState('');
    94→  const [editLessonType, setEditLessonType] = useState<'video' | 'link' | 'youtube' | 'document'>('video');
    95→  const [editLessonVideoUrl, setEditLessonVideoUrl] = useState('');
    96→  const [editLessonDesc, setEditLessonDesc] = useState('');
    97→  const [editLessonIsPreview, setEditLessonIsPreview] = useState(false);
    98→
    99→  // Reply form
   100→  const [replyTexts, setReplyTexts] = useState<Record<number, string>>({});
   101→
   102→  const toggleChapter = (chapterId: string) => {
   103→    setExpandedChapters((prev) => {
   104→      const next = new Set(prev);
   105→      if (next.has(chapterId)) next.delete(chapterId);
   106→      else next.add(chapterId);
   107→      return next;
   108→    });
   109→  };
   110→
   111→  const startEditChapter = (chapter: any) => {
   112→    setEditMode('chapter');
   113→    setEditId(chapter.id);
   114→    setEditChapterTitle(chapter.title || '');
   115→    setEditChapterDesc(chapter.description || '');
   116→    setAddMode(null);
   117→  };
   118→
   119→  const startEditLesson = (lesson: any) => {
   120→    setEditMode('lesson');
   121→    setEditId(lesson.id);
   122→    setEditLessonTitle(lesson.title || '');
   123→    setEditLessonType((lesson.lessonType || lesson.lesson_type || 'video') as any);
   124→    setEditLessonVideoUrl(lesson.videoUrl || lesson.video_url || '');
   125→    setEditLessonDesc(lesson.description || '');
   126→    setEditLessonIsPreview(!!lesson.isPreview);
   127→    setAddMode(null);
   128→  };
   129→
   130→  const cancelEdit = () => {
   131→    setEditMode(null);
   132→    setEditId('');
   133→  };
   134→
   135→  const handleCreateChapter = async () => {
   136→    if (!chapterTitle.trim()) { toast.error('Enter chapter title'); return; }
   137→    try {
   138→      const result = await createChapter({
   139→        title: chapterTitle.trim(),
   140→        subjectId,
   141→        description: chapterDesc.trim() || undefined,
   142→        sortOrder: subjectChapters.length + 1,
   143→      });
   144→      if (result) {
   145→        toast.success('Chapter created');
   146→        setChapterTitle('');
   147→        setChapterDesc('');
   148→        setAddMode(null);
   149→        refetchCurriculum();
   150→      } else {
   151→        toast.error('Failed to create chapter');
   152→      }
   153→    } catch (err: any) {
   154→      toast.error(err?.message || 'Failed to create chapter');
   155→    }
   156→  };
   157→
   158→  const handleUpdateChapter = async () => {
   159→    if (!editChapterTitle.trim()) { toast.error('Enter chapter title'); return; }
   160→    try {
   161→      const result = await updateChapter({
   162→        chapterId: editId,
   163→        title: editChapterTitle.trim(),
   164→        description: editChapterDesc.trim() || undefined,
   165→        subjectId,
   166→      });
   167→      if (result) {
   168→        toast.success('Chapter updated');
   169→        setEditMode(null);
   170→        setEditId('');
   171→        refetchCurriculum();
   172→      } else {
   173→        toast.error('Failed to update chapter');
   174→      }
   175→    } catch (err: any) {
   176→      toast.error(err?.message || 'Failed to update chapter');
   177→    }
   178→  };
   179→
   180→  const handleDeleteChapter = async (chapterId: string) => {
   181→    if (!confirm('Delete this chapter and all its lessons?')) return;
   182→    try {
   183→      const result = await deleteChapter({ chapterId });
   184→      if (result !== null) {
   185→        toast.success('Chapter deleted');
   186→        refetchCurriculum();
   187→      } else {
   188→        toast.error('Failed to delete chapter');
   189→      }
   190→    } catch (err: any) {
   191→      toast.error(err?.message || 'Failed to delete chapter');
   192→    }
   193→  };
   194→
   195→  const handleCreateLesson = async () => {
   196→    if (!lessonTitle.trim()) { toast.error('Enter lesson title'); return; }
   197→    if (!selectedChapterId) { toast.error('Select a chapter'); return; }
   198→    try {
   199→      const result = await createLesson({
   200→        title: lessonTitle.trim(),
   201→        chapterId: selectedChapterId,
   202→        subjectId,
   203→        lessonType,
   204→        videoUrl: lessonVideoUrl.trim() || undefined,
   205→        description: lessonDesc.trim() || undefined,
   206→        isPreview: lessonIsPreview,
   207→        sortOrder: (allLessons.filter((l: any) => (l.chapterId || l.chapter_id) === selectedChapterId).length) + 1,
   208→      });
   209→      if (result) {
   210→        toast.success('Lesson added');
   211→        setLessonTitle('');
   212→        setLessonVideoUrl('');
   213→        setLessonDesc('');
   214→        setLessonIsPreview(false);
   215→        setAddMode(null);
   216→        refetchCurriculum();
   217→      } else {
   218→        toast.error('Failed to add lesson');
   219→      }
   220→    } catch (err: any) {
   221→      toast.error(err?.message || 'Failed to add lesson');
   222→    }
   223→  };
   224→
   225→  const handleUpdateLesson = async () => {
   226→    if (!editLessonTitle.trim()) { toast.error('Enter lesson title'); return; }
   227→    try {
   228→      const result = await updateLesson({
   229→        lessonId: editId,
   230→        title: editLessonTitle.trim(),
   231→        description: editLessonDesc.trim() || undefined,
   232→        lessonType: editLessonType,
   233→        videoUrl: editLessonVideoUrl.trim() || undefined,
   234→        isPreview: editLessonIsPreview,
   235→      });
   236→      if (result) {
   237→        toast.success('Lesson updated');
   238→        setEditMode(null);
   239→        setEditId('');
   240→        refetchCurriculum();
   241→      } else {
   242→        toast.error('Failed to update lesson');
   243→      }
   244→    } catch (err: any) {
   245→      toast.error(err?.message || 'Failed to update lesson');
   246→    }
   247→  };
   248→
   249→  const handleDeleteLesson = async (lessonId: string) => {
   250→    if (!confirm('Delete this lesson?')) return;
   251→    try {
   252→      const result = await deleteLesson({ lessonId });
   253→      if (result !== null) {
   254→        toast.success('Lesson deleted');
   255→        refetchCurriculum();
   256→      } else {
   257→        toast.error('Failed to delete lesson');
   258→      }
   259→    } catch (err: any) {
   260→      toast.error(err?.message || 'Failed to delete lesson');
   261→    }
   262→  };
   263→
   264→  const handleReply = async (reviewId: number) => {
   265→    const text = replyTexts[reviewId]?.trim();
   266→    if (!text) return;
   267→    try {
   268→      const result = await replyReview({ reviewId, replyText: text });
   269→      if (result) {
   270→        toast.success('Reply posted');
   271→        setReplyTexts((prev) => ({ ...prev, [reviewId]: '' }));
   272→        refetchReviews();
   273→      } else {
   274→        toast.error('Failed to post reply');
   275→      }
   276→    } catch (err: any) {
   277→      toast.error(err?.message || 'Failed to post reply');
   278→    }
   279→  };
   280→
   281→  // Expand all chapters by default
   282→  useMemo(() => {
   283→    if (subjectChapters.length > 0 && expandedChapters.size === 0) {
   284→      setExpandedChapters(new Set(subjectChapters.map((ch: any) => ch.id)));
   285→    }
   286→  }, [subjectChapters.length]); // eslint-disable-line react-hooks/exhaustive-deps
   287→
   288→  if (courseLoading || subjectsLoading || curriculumLoading) {
   289→    return <LoadingSkeleton type="card" count={3} />;
   290→  }
   291→
   292→  if (courseError) {
   293→    return (
   294→      <div className="flex flex-col items-center justify-center py-20 space-y-4">
   295→        <p className="text-sm text-red-500">Failed to load course: {courseError}</p>
   296→        <GradientButton size="sm" onClick={() => refetchCurriculum()}>Retry</GradientButton>
   297→      </div>
   298→    );
   299→  }
   300→
   301→  if (!currentSubject) {
   302→    return (
   303→      <EmptyState
   304→        icon={BookOpen}
   305→        title="Subject not found"
   306→        description="This subject doesn't exist or has been removed."
   307→        actionLabel="Back to Curriculum"
   308→        onAction={() => navigate('course-curriculum', { courseId })}
   309→      />
   310→    );
   311→  }
   312→
   313→  const lessonTypeIcon: Record<string, typeof Video> = {
   314→    video: Video, link: LinkIcon, youtube: Video, document: FileText,
   315→  };
   316→  const lessonTypeLabel: Record<string, string> = {
   317→    video: 'Video', link: 'Link', youtube: 'YouTube', document: 'Document/PDF',
   318→  };
   319→  const lessonTypeColor: Record<string, string> = {
   320→    video: 'bg-neutral-100 text-neutral-700 dark:bg-neutral-800/30 dark:text-neutral-300',
   321→    link: 'bg-blue-100 text-blue-700 dark:bg-blue-900/30 dark:text-blue-400',
   322→    youtube: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   323→    document: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
   324→  };
   325→
   326→  return (
   327→    <div className="space-y-6">
   328→      {/* Header with Back */}
   329→      <motion.div initial={{ opacity: 0, y: 8 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }}>
   330→        <div className="flex items-center gap-3">
   331→          <button
   332→            onClick={() => navigate('course-curriculum', { courseId })}
   333→            className="w-10 h-10 rounded-xl bg-black/[0.03] dark:bg-white/[0.03] flex items-center justify-center text-muted-foreground hover:text-foreground hover:bg-black/[0.06] dark:hover:bg-white/[0.06] transition-colors"
   334→          >
   335→            <ArrowLeft className="w-5 h-5" />
   336→          </button>
   337→          <div className="flex-1 min-w-0">
   338→            <h1 className="text-2xl md:text-3xl font-extrabold text-foreground truncate">
   339→              {currentSubject.subjectName || currentSubject.subject_name || currentSubject.name || 'Subject'}
   340→            </h1>
   341→            <p className="text-muted-foreground mt-0.5 text-sm">{course?.title || ''}</p>
   342→          </div>
   343→          <GradientButton
   344→            size="sm"
   345→            icon={<Plus className="w-4 h-4" />}
   346→            onClick={() => { setAddMode('chapter'); setSelectedChapterId(''); setEditMode(null); }}
   347→          >
   348→            Chapter
   349→          </GradientButton>
   350→          <GradientButton
   351→            size="sm"
   352→            variant="ghost"
   353→            icon={<Plus className="w-4 h-4" />}
   354→            onClick={() => { setAddMode('lesson'); setSelectedChapterId(''); setEditMode(null); }}
   355→          >
   356→            Lesson
   357→          </GradientButton>
   358→        </div>
   359→      </motion.div>
   360→
   361→      {/* Add Chapter Form */}
   362→      <AnimatePresence>
   363→        {addMode === 'chapter' && (
   364→          <motion.div {...collapseExpand}>
   365→            <GlassCard className="p-5" glow>
   366→              <div className="flex items-center justify-between mb-4">
   367→                <h3 className="text-base font-bold gradient-text">New Chapter</h3>
   368→                <button onClick={() => setAddMode(null)} className="w-8 h-8 rounded-lg bg-muted/50 flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors">
   369→                  <X className="w-4 h-4" />
   370→                </button>
   371→              </div>
   372→              <div className="space-y-3">
   373→                <div>
   374→                  <label className="text-sm font-semibold text-foreground mb-1 block">Chapter Title *</label>
   375→                  <input
   376→                    type="text"
   377→                    value={chapterTitle}
   378→                    onChange={(e) => setChapterTitle(e.target.value)}
   379→                    placeholder="Enter chapter title"
   380→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60"
   381→                    autoFocus
   382→                  />
   383→                </div>
   384→                <div>
   385→                  <label className="text-sm font-semibold text-foreground mb-1 block">Description</label>
   386→                  <textarea
   387→                    value={chapterDesc}
   388→                    onChange={(e) => setChapterDesc(e.target.value)}
   389→                    placeholder="Optional chapter description"
   390→                    rows={2}
   391→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60 resize-none"
   392→                  />
   393→                </div>
   394→                <div className="flex justify-end gap-2">
   395→                  <GradientButton variant="ghost" size="sm" onClick={() => setAddMode(null)}>Cancel</GradientButton>
   396→                  <GradientButton size="sm" onClick={handleCreateChapter} loading={creatingChapter} disabled={!chapterTitle.trim()} icon={<Save className="w-3.5 h-3.5" />}>
   397→                    Create Chapter
   398→                  </GradientButton>
   399→                </div>
   400→              </div>
   401→            </GlassCard>
   402→          </motion.div>
   403→        )}
   404→      </AnimatePresence>
   405→
   406→      {/* Edit Chapter Form */}
   407→      <AnimatePresence>
   408→        {editMode === 'chapter' && (
   409→          <motion.div {...collapseExpand}>
   410→            <GlassCard className="p-5" glow>
   411→              <div className="flex items-center justify-between mb-4">
   412→                <h3 className="text-base font-bold gradient-text">Edit Chapter</h3>
   413→                <button onClick={cancelEdit} className="w-8 h-8 rounded-lg bg-muted/50 flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors">
   414→                  <X className="w-4 h-4" />
   415→                </button>
   416→              </div>
   417→              <div className="space-y-3">
   418→                <div>
   419→                  <label className="text-sm font-semibold text-foreground mb-1 block">Chapter Title *</label>
   420→                  <input
   421→                    type="text"
   422→                    value={editChapterTitle}
   423→                    onChange={(e) => setEditChapterTitle(e.target.value)}
   424→                    placeholder="Enter chapter title"
   425→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60"
   426→                    autoFocus
   427→                  />
   428→                </div>
   429→                <div>
   430→                  <label className="text-sm font-semibold text-foreground mb-1 block">Description</label>
   431→                  <textarea
   432→                    value={editChapterDesc}
   433→                    onChange={(e) => setEditChapterDesc(e.target.value)}
   434→                    placeholder="Optional chapter description"
   435→                    rows={2}
   436→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60 resize-none"
   437→                  />
   438→                </div>
   439→                <div className="flex justify-end gap-2">
   440→                  <GradientButton variant="ghost" size="sm" onClick={cancelEdit}>Cancel</GradientButton>
   441→                  <GradientButton size="sm" onClick={handleUpdateChapter} loading={updatingChapter} disabled={!editChapterTitle.trim()} icon={<Save className="w-3.5 h-3.5" />}>
   442→                    Save Changes
   443→                  </GradientButton>
   444→                </div>
   445→              </div>
   446→            </GlassCard>
   447→          </motion.div>
   448→        )}
   449→      </AnimatePresence>
   450→
   451→      {/* Add Lesson Form */}
   452→      <AnimatePresence>
   453→        {addMode === 'lesson' && (
   454→          <motion.div {...collapseExpand}>
   455→            <GlassCard className="p-5" glow>
   456→              <div className="flex items-center justify-between mb-4">
   457→                <h3 className="text-base font-bold gradient-text">New Lesson</h3>
   458→                <button onClick={() => setAddMode(null)} className="w-8 h-8 rounded-lg bg-muted/50 flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors">
   459→                  <X className="w-4 h-4" />
   460→                </button>
   461→              </div>
   462→              <div className="space-y-3">
   463→                <div>
   464→                  <label className="text-sm font-semibold text-foreground mb-1 block">Chapter *</label>
   465→                  <select
   466→                    value={selectedChapterId}
   467→                    onChange={(e) => setSelectedChapterId(e.target.value)}
   468→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input"
   469→                  >
   470→                    <option value="">Select chapter</option>
   471→                    {subjectChapters.map((ch: any) => (
   472→                      <option key={ch.id} value={ch.id}>{ch.title}</option>
   473→                    ))}
   474→                  </select>
   475→                  {subjectChapters.length === 0 && (
   476→                    <p className="text-xs text-amber-600 dark:text-amber-400 mt-1">Create a chapter first before adding lessons.</p>
   477→                  )}
   478→                </div>
   479→                <div>
   480→                  <label className="text-sm font-semibold text-foreground mb-1 block">Lesson Title *</label>
   481→                  <input
   482→                    type="text"
   483→                    value={lessonTitle}
   484→                    onChange={(e) => setLessonTitle(e.target.value)}
   485→                    placeholder="Enter lesson title"
   486→                    className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60"
   487→                  />
   488→                </div>
   489→                <div className="grid grid-cols-2 gap-3">
   490→                  <div>
   491→                    <label className="text-sm font-semibold text-foreground mb-1 block">Type</label>
   492→                    <select
   493→                      value={lessonType}
   494→                      onChange={(e) => setLessonType(e.target.value as any)}
   495→                      className="w-full px-4 py-2.5 rounded-xl text-sm premium-input"
   496→                    >
   497→                      <option value="video">Video</option>
   498→                      <option value="link">Link / Live</option>
   499→                      <option value="youtube">YouTube</option>
   500→                      <option value="document">Document / PDF / Note</option>
   501→                    </select>
   502→                  </div>
   503→                  <div>
   504→                    <label className="text-sm font-semibold text-foreground mb-1 block">Video/Link URL</label>
   505→                    <input
   506→                      type="url"
   507→                      value={lessonVideoUrl}
   508→                      onChange={(e) => setLessonVideoUrl(e.target.value)}
   509→                      placeholder="https://..."
   510→                      className="w-full px-4 py-2.5 rounded-xl text-sm premium-input placeholder:text-muted-foreground/60"
   511→                    />
   512→                  </div>
