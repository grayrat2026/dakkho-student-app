    32тЖТ    32тЖТ    const video = document.createElement('video');
    33тЖТ    33тЖТ    video.preload = 'metadata';
    34тЖТ    34тЖТ    video.onloadedmetadata = () => {
    35тЖТ    35тЖТ      resolve(Math.round(video.duration));
    36тЖТ    36тЖТ      URL.revokeObjectURL(video.src);
