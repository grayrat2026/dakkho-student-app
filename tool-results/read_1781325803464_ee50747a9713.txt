    47→// Success confetti particles
    48→function ConfettiParticles() {
    49→  const particles = Array.from({ length: 30 }, (_, i) => ({
    50→    id: i,
    51→    x: Math.random() * 300 - 150,
    52→    y: Math.random() * 300 - 150,
    53→    rotation: Math.random() * 360,
    54→    scale: Math.random() * 0.5 + 0.5,
    55→    color: ['#0ea5e9', '#10b981', '#f59e0b', '#ef4444', '#8b5cf6', '#ec4899'][Math.floor(Math.random() * 6)],
    56→    delay: Math.random() * 0.5,
    57→  }));
    58→
    59→  return (
    60→    <div className="absolute inset-0 pointer-events-none overflow-hidden">
    61→      {particles.map((p) => (
    62→        <motion.div
    63→          key={p.id}
    64→          className="absolute left-1/2 top-1/2 w-2 h-2 rounded-sm"
    65→          style={{ backgroundColor: p.color }}
    66→          initial={{ x: 0, y: 0, scale: 0, rotate: 0, opacity: 1 }}
    67→          animate={{
    68→            x: p.x,
    69→            y: p.y,
    70→            scale: p.scale,
    71→            rotate: p.rotation,
    72→            opacity: 0,
    73→          }}
    74→          transition={{
    75→            duration: 1.5,
    76→            delay: p.delay,
    77→            ease: [0.4, 0, 0.2, 1],
    78→          }}
    79→        />
    80→      ))}
    81→    </div>
    82→  );
    83→}
