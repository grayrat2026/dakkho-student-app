    77→async function verifyCourseOwnership(env: any, courseId: string, instructorId: string): Promise<boolean> {
    78→  const course = await env.DB.prepare('SELECT instructor_id FROM courses WHERE id = ?').bind(courseId).first<{ instructor_id: string }>();
    79→  if (!course) return false;
    80→  if (course.instructor_id === instructorId) return true;
    81→  // Also check course_instructors junction table
    82→  const junction = await env.DB.prepare('SELECT id FROM course_instructors WHERE course_id = ? AND instructor_id = ?').bind(courseId, instructorId).first();
    83→  return !!junction;
    84→}

    90→// POST /auth/login — Instructor login (public, no auth middleware)
    91→instructorRoutes.post('/auth/login', async (c) => {
    92→  try {
    93→    const { email, password } = await c.req.json<{ email: string; password: string }>();
    94→
    95→    if (!email || !password) {
    96→      return c.json({ error: 'Email and password are required' }, 400);
    97→    }
    98→
    99→    // Authenticate user via D1
   100→    const authResult = await authenticateUser(c.env, email, password);
   101→
   102→    if (!authResult.success) {
   103→      return c.json({ error: 'Invalid email or password' }, 401);
   104→    }
   105→
   106→    const userId = authResult.userId!;
   107→    const userEmail = authResult.userEmail!;
   108→    const userName = authResult.userName!;
   109→    const userRole = authResult.userRole || 'student';
   110→    const avatarUrl = authResult.avatarUrl || '';
   111→
   112→    // Only allow instructor role
   113→    if (userRole !== 'instructor') {
   114→      return c.json({ error: 'This login is for instructors only. Please use the student or admin portal.' }, 403);
   115→    }
   116→
   117→    // Create instructor session
   118→    const token = await createInstructorSession(
   119→      c.env,
   120→      userId,
   121→      userEmail,
   122→      userName,
   123→      c.req.header('x-forwarded-for') || c.req.header('x-real-ip') || 'unknown',
   124→      c.req.header('user-agent') || 'unknown',
   125→      avatarUrl
   126→    );
   127→
   128→    // Also fetch the full instructor profile
   129→    let profile: Record<string, unknown> | null = null;
   130→    try {
   131→      const row = await c.env.DB.prepare(
   132→        'SELECT * FROM instructors WHERE id = ?'
   133→      ).bind(userId).first();
   134→      if (row) {
   135→        profile = formatInstructorRow(row as Record<string, unknown>);
   136→      }
   137→    } catch {}
   138→
   139→    return c.json({
   140→      success: true,
   141→      token,
   142→      user: {
   143→        id: userId,
   144→        email: userEmail,
   145→        name: userName,
   146→        role: 'instructor',
   147→        avatarUrl,
   148→        ...(profile || {}),
   149→      },
   150→    });
   151→  } catch (error) {
   152→    const message = getErrorMessage(error);
   153→    console.error('Instructor login error:', error);
   154→    return c.json({ error: message }, 401);
   155→  }
   156→});

   220→// POST /auth/reset-password — Reset password with OTP (public)
   221→instructorRoutes.post('/auth/reset-password', async (c) => {
   222→  try {
   223→    const { email, otp, password } = await c.req.json<{ email: string; otp: string; password: string }>();
   224→    if (!email || !otp || !password) {
   225→      return c.json({ error: 'email, otp, and password are required' }, 400);
   226→    }
   227→    if (password.length < 8) {
   228→      return c.json({ error: 'Password must be at least 8 characters' }, 400);
   229→    }
   230→
   231→    // Verify OTP
   232→    const otpRecord = await c.env.DB.prepare(
   233→      "SELECT id, expires_at, verified FROM otp_codes WHERE target = ? AND code = ? AND type = 'password_reset' ORDER BY created_at DESC LIMIT 1"
   234→    ).bind(email, otp).first<{ id: number; expires_at: string; verified: number }>();
   235→
   236→    if (!otpRecord || otpRecord.verified || new Date(otpRecord.expires_at) < new Date()) {
   237→      return c.json({ error: 'Invalid or expired OTP code' }, 400);
   238→    }
   239→
   240→    // Mark OTP as used
   241→    await c.env.DB.prepare('UPDATE otp_codes SET verified = 1 WHERE id = ?').bind(otpRecord.id).run();
   242→
   243→    // Find user
   244→    const user = await c.env.DB.prepare('SELECT id FROM users WHERE email = ?').bind(email).first<{ id: string }>();
   245→    if (!user) return c.json({ error: 'User not found' }, 404);
   246→
   247→    // Update password
   248→    const passwordHash = await hashPassword(password);
   249→    await c.env.DB.prepare(
   250→      'UPDATE users SET password_hash = ?, password_migrated = 1, updated_at = ? WHERE id = ?'
   251→    ).bind(passwordHash, new Date().toISOString(), user.id).run();
   252→
   253→    // Invalidate all sessions
   254→    try { await c.env.DB.prepare('UPDATE instructor_sessions SET is_active = 0 WHERE user_id = ?').bind(user.id).run(); } catch {}
   255→
   256→    return c.json({ success: true, message: 'Password has been reset successfully. Please login with your new password.' });
   257→  } catch (error) {
   258→    return c.json({ error: getErrorMessage(error) }, 500);
   259→  }
   260→});

   299→// POST /change-password — Change instructor password (instructorAuth ONLY)
   300→instructorRoutes.post('/change-password', instructorAuthMiddleware, async (c) => {
   301→  try {
   302→    const instructorId = c.get('instructorId');
   303→    const instructorEmail = c.get('instructorEmail');
   304→    const { current_password, new_password } = await c.req.json<{ current_password: string; new_password: string }>();
   305→
   306→    if (!current_password || !new_password) {
   307→      return c.json({ error: 'Current and new password are required' }, 400);
   308→    }
   309→
   310→    if (new_password.length < 8) {
   311→      return c.json({ error: 'New password must be at least 8 characters' }, 400);
   312→    }
   313→
   314→    // Verify current password using D1 lookup + verifyPassword
   315→    const user = await c.env.DB.prepare(
   316→      'SELECT id, email, password_hash, password_migrated FROM users WHERE id = ?'
   317→    ).bind(instructorId).first<{ id: string; email: string; password_hash: string | null; password_migrated: number }>();
   318→
   319→    if (!user) {
   320→      return c.json({ error: 'User not found' }, 404);
   321→    }
   322→
   323→    if (user.password_hash && user.password_migrated === 1) {
   324→      // Fully migrated — verify with D1 password hash
   325→      const valid = await verifyPassword(current_password, user.password_hash);
   326→      if (!valid) {
   327→        return c.json({ error: 'Current password is incorrect' }, 400);
   328→      }
   329→    } else {
   330→      // Not migrated yet — use authenticateUser which falls back to Appwrite
   331→      const authResult = await authenticateUser(c.env, instructorEmail, current_password);
   332→      if (!authResult.success) {
   333→        return c.json({ error: 'Current password is incorrect' }, 400);
   334→      }
   335→    }
   336→
   337→    // Hash the new password and update in D1
   338→    const newPasswordHash = await hashPassword(new_password);
   339→    await c.env.DB.prepare(
   340→      'UPDATE users SET password_hash = ?, password_migrated = 1, updated_at = ? WHERE id = ?'
   341→    ).bind(newPasswordHash, new Date().toISOString(), instructorId).run();
   342→
   343→    return c.json({ success: true, message: 'Password changed successfully' });
   344→  } catch (error) {
   345→    return c.json({ error: getErrorMessage(error) }, 500);
   346→  }
   347→});

   353→// GET /dashboard — Instructor dashboard stats
   354→instructorRoutes.get('/dashboard', instructorOrAdminMiddleware, async (c) => {
   355→  try {
   356→    const authRole = c.get('authRole');
   357→    let instructorId: string;
   358→
   359→    if (authRole === 'admin') {
   360→      instructorId = c.req.query('instructorId') || '';
   361→      if (!instructorId) {
   362→        return c.json({ error: 'instructorId query param required for admin access' }, 400);
   363→      }
   364→    } else {
   365→      instructorId = c.get('instructorId');
   366→    }
   367→
   368→    // Course count
   369→    let courseCount = 0;
   370→    try {
   371→      const cc = await c.env.DB.prepare('SELECT COUNT(*) as total FROM courses WHERE instructor_id = ?').bind(instructorId).first<{ total: number }>();
   372→      courseCount = cc?.total || 0;
   373→    } catch {}
   374→
   375→    // Total students (from enrollments across all instructor courses)
   376→    let totalStudents = 0;
   377→    try {
   378→      const ts = await c.env.DB.prepare(
   379→        'SELECT COUNT(DISTINCT user_id) as total FROM enrollments WHERE course_id IN (SELECT id FROM courses WHERE instructor_id = ?)'
   380→      ).bind(instructorId).first<{ total: number }>();
   381→      totalStudents = ts?.total || 0;
   382→    } catch {}
   383→
   384→    // Average rating
   385→    let avgRating = 0;
   386→    let totalReviews = 0;
   387→    try {
   388→      const rs = await c.env.DB.prepare(
   389→        'SELECT AVG(rating) as avg, COUNT(*) as count FROM instructor_reviews WHERE instructor_id = ?'
   390→      ).bind(instructorId).first<{ avg: number; count: number }>();
   391→      avgRating = rs?.avg ? Math.round(rs.avg * 10) / 10 : 0;
   392→      totalReviews = rs?.count || 0;
   393→    } catch {}
   394→
   395→    // Total revenue (from payments for instructor's courses)
   396→    let totalRevenue = 0;
   397→    try {
   398→      const rev = await c.env.DB.prepare(
   399→        "SELECT COALESCE(SUM(amount), 0) as total FROM payments WHERE course_id IN (SELECT id FROM courses WHERE instructor_id = ?) AND status = 'completed'"
   400→      ).bind(instructorId).first<{ total: number }>();
   401→      totalRevenue = rev?.total || 0;
   402→    } catch {}
   403→
   404→    // Upcoming classes count
   405→    let upcomingClasses = 0;
   406→    try {
   407→      const uc = await c.env.DB.prepare(
   408→        "SELECT COUNT(*) as total FROM live_class_schedules WHERE instructor_id = ? AND scheduled_at > datetime('now') AND is_active = 1"
   409→      ).bind(instructorId).first<{ total: number }>();
   410→      upcomingClasses = uc?.total || 0;
   411→    } catch {}
   412→
   413→    return c.json({
   414→      success: true,
   415→      dashboard: {
   416→        courseCount,
   417→        totalStudents,
   418→        avgRating,
   419→        totalReviews,
   420→        totalRevenue,
   421→        upcomingClasses,
   422→      },
   423→    });
   424→  } catch (error) {
   425→    return c.json({ error: getErrorMessage(error) }, 500);
   426→  }
   427→});

   465→// PUT /profile — Update instructor profile (name, bio, avatar, etc.)
   466→instructorRoutes.put('/profile', instructorOrAdminMiddleware, async (c) => {
   467→  try {
   468→    const authRole = c.get('authRole');
   469→    let instructorId: string;
   470→
   471→    if (authRole === 'admin') {
   472→      instructorId = c.req.query('instructorId') || '';
   473→      if (!instructorId) {
   474→        return c.json({ error: 'instructorId query param required for admin access' }, 400);
   475→      }
   476→    } else {
   477→      instructorId = c.get('instructorId');
   478→    }
   479→
   480→    const body = await c.req.json();
   481→
   482→    // Map camelCase field names to snake_case D1 columns
   483→    const fieldMapping: Record<string, string> = {
   484→      name: 'name',
   485→      bio: 'bio',
   486→      avatar: 'avatar_url',
   487→      avatarUrl: 'avatar_url',
   488→      specialization: 'specialization',
   489→      phone: 'phone',
   490→      department: 'department',
   491→    };
   492→
   493→    const setClauses: string[] = [];
   494→    const params: unknown[] = [];
   495→
   496→    for (const [bodyField, dbColumn] of Object.entries(fieldMapping)) {
   497→      if (body[bodyField] !== undefined) {
   498→        setClauses.push(`${dbColumn} = ?`);
   499→        params.push(body[bodyField]);
   500→      }
   501→    }
   502→
   503→    if (setClauses.length === 0) {
   504→      return c.json({ error: 'No valid fields to update' }, 400);
   505→    }
   506→
   507→    // Always update updated_at
   508→    setClauses.push('updated_at = ?');
   509→    params.push(new Date().toISOString());
   510→
   511→    params.push(instructorId);
   512→
   513→    await c.env.DB.prepare(
   514→      `UPDATE instructors SET ${setClauses.join(', ')} WHERE id = ?`
   515→    ).bind(...params).run();
   516→
   517→    // If name was updated, also update the D1 session
   518→    if ((body.name !== undefined) && authRole === 'instructor') {
   519→      try {
   520→        await c.env.DB.prepare(
   521→          'UPDATE instructor_sessions SET name = ? WHERE user_id = ? AND is_active = 1'
   522→        ).bind(String(body.name), instructorId).run();
   523→      } catch {}
   524→    }
   525→
   526→    // If avatar was updated, also update the D1 session
   527→    if ((body.avatarUrl !== undefined || body.avatar !== undefined) && authRole === 'instructor') {
   528→      try {
   529→        const avatarVal = body.avatarUrl || body.avatar;
   530→        await c.env.DB.prepare(
   531→          'UPDATE instructor_sessions SET avatar_url = ? WHERE user_id = ? AND is_active = 1'
   532→        ).bind(String(avatarVal), instructorId).run();
   533→      } catch {}
   534→    }
   535→
   536→    // Fetch and return the updated profile
   537→    const updatedRow = await c.env.DB.prepare(
   538→      'SELECT * FROM instructors WHERE id = ?'
   539→    ).bind(instructorId).first();
   540→
   541→    const profile = formatInstructorRow(updatedRow as Record<string, unknown>);
   542→
   543→    return c.json({ success: true, profile });
   544→  } catch (error) {
   545→    return c.json({ error: getErrorMessage(error) }, 500);
   546→  }
   547→});

   549→// POST /profile/avatar — Upload instructor avatar to R2
   550→instructorRoutes.post('/profile/avatar', instructorOrAdminMiddleware, async (c) => {
   551→  try {
